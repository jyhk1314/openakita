	int ConvertOperator(sWhereClause &tWhereClause)
	{
		switch (tWhereClause.Operator)
		{
			case _IN:
			{
				tWhereClause.Operator = NOT_IN;
				break;
			}
			case LIKE:
			{
				tWhereClause.Operator = NOT_LIKE;
				break;
			}
			default:
			{
                MDB_ERR("Only support 'not in' or 'not like'!");
                return -1; 
			}
		}
		return 0;
	}

    int GenExpr(TExpr *pExpr, string &sExpr, string &sErrMsg)
    {//留待下个版本细分，不然太难看
        int  nRet = 0; 
        char szDigit[64]  = ""; 
        char szErr[256] = ""; 
        if(NULL  ==  pExpr) return 0; 
        switch(pExpr->nOper)
        {
            case TK_INTEGER:
                snprintf(szDigit, sizeof(szDigit), "%lld",  pExpr->u.nValue); 
                sExpr.append(szDigit); 
                break; 
            case TK_FLOAT:
                snprintf(szDigit, sizeof(szDigit),  "%f",  pExpr->u.dValue); 
                sExpr.append(szDigit); 
                break; 
            case TK_STRING:
                sExpr.append("'").append(pExpr->u.zToken).append("'"); 
                break; 
            case TK_COLON: // 绑定变量
                sExpr.append(":"); 
                sExpr.append(pExpr->u.zToken); 
                break; 
            case TK_COUNT:
                sExpr.append("COUNT_"); 
                break; 
            case TK_SYSDATE:
                sExpr.append("SYSDATE"); 
                break; 
            case TK_TO_DATE:
            case TK_TO_CHAR:
                //指针在解析的时候已经判断
                assert(pExpr->pList); 
                pExpr = pExpr->pList->t[0].pExpr; 
                assert(pExpr); 
                if(TK_ID == pExpr->nOper)
                {
                    sExpr.append(pExpr->u.zToken); 
                }
                else if(TK_STRING == pExpr->nOper)
                {
                    sExpr.append("'").append(pExpr->u.zToken).append("'"); 
                }
                else if(TK_COLON == pExpr->nOper)
                {
                    sExpr.append(":"); 
                    sExpr.append(pExpr->u.zToken); 
                }
                else
                {
                    snprintf(szErr, sizeof(szErr), "函数表达式中该操作符[%d]尚不支持",  pExpr->nOper); 
                    sErrMsg = szErr; 
                    MDB_ERR("%s", sErrMsg.c_str());
                    return -1; 
                    break; 
                }
                break; 
            case TK_PLUS:
                // mdy by jyhk at 2024-06-06, reason: 表别名改造
				if (pExpr->pLeft->nOper == TK_DOT)
				{
					nRet = GenExpr(pExpr->pLeft->pRight, sExpr, sErrMsg); 
					if(0 != nRet)
					{
						MDB_CALL_ERR("GenExpr()"); 
						return nRet; 
					}
				}
				else
				{
					nRet = GenExpr(pExpr->pLeft, sExpr, sErrMsg); 
					if(0 != nRet)
					{
						MDB_CALL_ERR("GenExpr()"); 
						return nRet; 
					}
				}
                
                sExpr.append("+"); 
                nRet = GenExpr(pExpr->pRight, sExpr, sErrMsg); 
                if(0 != nRet)
                {
                    MDB_CALL_ERR("GenExpr()"); 
                    return nRet; 
                }
                //sExpr.append(")"); 
                break; 
            case TK_MINUS:
                // mdy by jyhk at 2024-06-06, reason: 表别名改造
				if (pExpr->pLeft->nOper == TK_DOT)
				{
					nRet = GenExpr(pExpr->pLeft->pRight, sExpr, sErrMsg); 
					if(0 != nRet)
					{
						MDB_CALL_ERR("GenExpr()"); 
						return nRet; 
					}
				}
				else
				{
					nRet = GenExpr(pExpr->pLeft, sExpr, sErrMsg); 
					if(0 != nRet)
					{
						MDB_CALL_ERR("GenExpr()"); 
						return nRet; 
					}
				}

                sExpr.append("-"); 
                nRet = GenExpr(pExpr->pRight, sExpr, sErrMsg); 
                if(0 != nRet)
                {
                    MDB_CALL_ERR("GenExpr()"); 
                    return nRet; 
                }
                //sExpr.append(")"); 
                break; 
            case TK_UMINUS:
                sExpr.append("-"); 
                //through,  no break
            case TK_UPLUS:
                assert(pExpr->pLeft); 
                if(TK_INTEGER  != pExpr->pLeft->nOper &&  TK_FLOAT  != pExpr->pLeft->nOper)
                {
                    sErrMsg = "不合法的字符["; 
                    sErrMsg.append(pExpr->pLeft->u.zToken).append("]"); 
                    MDB_ERR("%s", sErrMsg.c_str()); 
                    return -1; 
                }
                //sExpr.append("+"); //+ 号无需填
                nRet = GenExpr(pExpr->pLeft, sExpr, sErrMsg); 
                if(0 != nRet)
                {
                    MDB_CALL_ERR("GenExpr()"); 
                    return nRet; 
                }
                break; 
            case TK_STAR:
            case TK_SLASH:           	
            case TK_WILDCARD:
                sErrMsg = "不支持的表达式"; 
                sErrMsg.append("[").append("*/?").append("]");
                MDB_ERR("%s", sErrMsg.c_str()); 
                return -1; 
                break;
            default:
                if(pExpr->u.zToken)
                {
                    sExpr.append(pExpr->u.zToken); 
                }
                break; 
        }

        return nRet; 
    }

	int SelectColumnCheck(ZmdbParseSqlStruct &tResult)
	{
		if(true == tResult.tSelectType.bSum && true == tResult.tSelectType.bCount )
		{
			MDB_ERR("[count] and [sum] can't be used simultaneously!"); 
			return -1; 
		}
		
		if(true == tResult.tSelectType.bMax && true == tResult.tSelectType.bMin )
		{
			MDB_ERR("[max] and [min] can't be used simultaneously!"); 
			return -1; 
		}
		
		if(true == tResult.tSelectType.bSum || true == tResult.tSelectType.bCount)
		{
			if(true == tResult.tSelectType.bMax || true == tResult.tSelectType.bMin )
			{
				MDB_ERR("[max] or [min] and [count] or [sum] can't be used simultaneously!"); 
				return -1; 
			}
		}

		if(true == tResult.tSelectType.bGroupby)
		{
			if (true == tResult.tSelectType.bAll)
			{
				MDB_ERR("[GroupBy] and [*] can't be used simultaneously!"); 
				return -1; 
			}
		}
		else if (true == tResult.tSelectType.bSort)
		{
			if(true == tResult.tSelectType.bCount && false == tResult.tSelectType.bDistinct)
			{
				MDB_ERR("排序时不支持使用Count!");
				return -1; 
			}

			if(true == tResult.tSelectType.bSum)
			{
				MDB_ERR("排序时不支持使用Sum!");
				return -1; 
			}
		}

		// add by jyhk at 2024-06-07, reason: 单表查询别名全域检测 
		if (tResult.fromClauseCount == 1)
		{
			// 稽核条件是否带别名
			for (int i = 0; i < tResult.whereClauseCount; i++)
			{
				if (tResult.whereClause[i].TableAliasName[0] != '\0')
				{
					if (TZmdbStrFunc::StrCmpNoCase(tResult.fromClause[0].AliasName, tResult.whereClause[i].TableAliasName) != 0)
					{
						MDB_ERR("操作字段中表别名与操作表不匹配");
						return -1;
					}
				}
			}
		}

		return 0;
	}

    int SelectColumnMap(TExprList *pColList, TExprList *pGroupbyExprList,ZmdbParseSqlStruct &tResult, string &sErrMsg, TZmdbOrderBy* pOrderBy, int& OrderByCounts, TZmdbLimit* pLimit, int iLevel)
    {
        int nRet = 0; 
        char szErr[256] = ""; 
        assert(pColList); 
        for(int i = 0;  i < pColList->nExpr;  ++i)
        {
            TExpr *pExpr = pColList->t[i].pExpr;
            assert(pExpr); 

            if(tResult.columnCount  >=  (MAX_COLUMN_COUNT))
            {
                sErrMsg =  "查询字段数超过限制"; 
                MDB_ERR("%s", sErrMsg.c_str()); 
                return -1;
            }
            switch(pExpr->nOper)
            {
                case TK_DOT:			
					tResult.columnOper[tResult.columnCount].Clear();
					if (pExpr->pLeft  == NULL || pExpr->pLeft->nOper != TK_ID 
						|| pExpr->pRight == NULL || pExpr->pRight->nOper != TK_ID)
					{
						MDB_CALL_ERR("Column token used tk_dot'.', left/right oper must be id!"); 
						return nRet; 
					}
                    strncpy(tResult.columnOper[tResult.columnCount].TableAliasName, pExpr->pLeft->u.zToken,  ZMDB_MAX_NAME_LEN-1); 
                    tResult.columnOper[tResult.columnCount].TableAliasName[ZMDB_MAX_NAME_LEN - 1] = '\0'; 
                    strncpy(tResult.columnOper[tResult.columnCount].ColumnName, pExpr->pRight->u.zToken,  ZMDB_MAX_NAME_LEN-1); 
                    tResult.columnOper[tResult.columnCount].ColumnName[ZMDB_MAX_NAME_LEN - 1] = '\0'; 
					
					//add by lijing 20230110: 获取字段别名
					if(pColList->t[i].pName != NULL)
					{
						strncpy(tResult.columnOper[tResult.columnCount].AliasName, pColList->t[i].pName,  ZMDB_MAX_NAME_LEN-1); 
                    	tResult.columnOper[tResult.columnCount].AliasName[ZMDB_MAX_NAME_LEN - 1] = '\0'; 
					}
					
                    tResult.columnCount++;

                    break; 
                case TK_COUNT:			
					tResult.columnOper[tResult.columnCount].Clear();
					tResult.columnOper[tResult.columnCount].bCount = true;
					strncpy(tResult.columnOper[tResult.columnCount].ColumnName, "count", ZMDB_MAX_NAME_LEN); 
					
					//add by lijing 20230110: 获取字段别名
					if(pColList->t[i].pName != NULL)
					{
						strncpy(tResult.columnOper[tResult.columnCount].AliasName, pColList->t[i].pName,  ZMDB_MAX_NAME_LEN-1); 
                    	tResult.columnOper[tResult.columnCount].AliasName[ZMDB_MAX_NAME_LEN - 1] = '\0'; 
					}
					
					tResult.columnCount++;
					tResult.tSelectType.bCount = true;
					if (pExpr->pList != NULL)
					{
						if (pExpr->pList->t[0].pExpr->nOper == TK_DISTINCT)
						{
							//函数嵌套count(distinct id)，
							nRet = SelectColumnMap(pExpr->pList, pGroupbyExprList, tResult, sErrMsg, pOrderBy, OrderByCounts, pLimit, iLevel++);
							if(0 != nRet)
							{
								MDB_CALL_ERR("SelectColumnMap()"); 
								return nRet; 
							}
						}
					}
                    break; 
				case TK_DISTINCT:
					if ((tResult.columnCount != 0 && tResult.tSelectType.bCount == false) 
						|| (tResult.columnCount != 1 && tResult.tSelectType.bCount == true))
					{
                        snprintf(szErr, sizeof(szErr), "Select Distinct must be before Field!"); 
                        sErrMsg = szErr; 
                        MDB_ERR("%s", sErrMsg.c_str()); 
                        return -1; 
					}

					if (pGroupbyExprList == NULL)
					{
						tResult.tSelectType.bDistinct = true;
					}
					break; 
				case TK_ALL:					
					tResult.columnOper[tResult.columnCount].Clear();
					if (true == tResult.tSelectType.bDistinct)
					{
						snprintf(szErr, sizeof(szErr),	"Select Distinct Support Only Named Fields!."); 
						sErrMsg = szErr; 
						MDB_ERR("%s", sErrMsg.c_str()); 
						return -1; 
					}
					tResult.columnCount = 1; 
					tResult.tSelectType.bAll = true;
					return 0;  //直接返回，让外部去填充
					break; 
				case TK_MAX:
                    if(NULL == pExpr->pList 
                        || NULL == pExpr->pList->t
                        || NULL == pExpr->pList->t->pExpr
                        || NULL == pExpr->pList->t->pExpr->u.zToken
                        )
                    {
                        snprintf(szErr, sizeof(szErr),  "函数[max]未指定列名."); 
                        sErrMsg = szErr; 
                        MDB_ERR("%s", sErrMsg.c_str()); 
                        return -1; 
                    }
					
					if (true == tResult.tSelectType.bDistinct)
					{
                        snprintf(szErr, sizeof(szErr),  "函数[max]不支持和[distinct]一起使用."); 
                        sErrMsg = szErr; 
                        MDB_ERR("%s", sErrMsg.c_str()); 
                        return -1; 
					}
					tResult.columnOper[tResult.columnCount].Clear();
                    strncpy(tResult.columnOper[tResult.columnCount].ColumnName, pExpr->pList->t->pExpr->u.zToken, ZMDB_MAX_NAME_LEN - 1);
                    tResult.columnOper[tResult.columnCount].ColumnName[ZMDB_MAX_NAME_LEN - 1] = '\0';
					
					//add by lijing 20230110: 获取字段别名
					if(pColList->t[i].pName != NULL)
					{
						strncpy(tResult.columnOper[tResult.columnCount].AliasName, pColList->t[i].pName,  ZMDB_MAX_NAME_LEN-1); 
                    	tResult.columnOper[tResult.columnCount].AliasName[ZMDB_MAX_NAME_LEN - 1] = '\0'; 
					}
					
					tResult.columnOper[tResult.columnCount].bMax = true;
                    tResult.columnCount++;

					strncpy(pOrderBy[0].sParamName, pExpr->pList->t->pExpr->u.zToken, ZMDB_MAX_NAME_LEN - 1); 
					pOrderBy[0].sParamName[ZMDB_MAX_NAME_LEN - 1] = '\0';
					pOrderBy[0].bIsAsc = false; 
					OrderByCounts = 1;

					pLimit->iLeftLimitNum = 0;
					pLimit->iRightLimitNum = 1;

					tResult.tSelectType.bMax = true;
					tResult.tSelectType.bSort = true;					
					tResult.tSelectType.bLimit = true;
                    break; 
				case TK_MIN:
                    if(NULL == pExpr->pList 
                        || NULL == pExpr->pList->t
                        || NULL == pExpr->pList->t->pExpr
                        || NULL == pExpr->pList->t->pExpr->u.zToken
                        )
                    {
                        snprintf(szErr, sizeof(szErr), "函数[min]未指定列名."); 
                        sErrMsg = szErr; 
                        MDB_ERR("%s", sErrMsg.c_str()); 
                        return -1; 
                    }
					
					if (true == tResult.tSelectType.bDistinct)
					{
                        snprintf(szErr, sizeof(szErr),  "函数[min]不支持和[distinct]一起使用."); 
                        sErrMsg = szErr; 
                        MDB_ERR("%s", sErrMsg.c_str()); 
                        return -1; 
					}
					tResult.columnOper[tResult.columnCount].Clear();
                    strncpy(tResult.columnOper[tResult.columnCount].ColumnName, pExpr->pList->t->pExpr->u.zToken, ZMDB_MAX_NAME_LEN - 1); 
					tResult.columnOper[tResult.columnCount].ColumnName[ZMDB_MAX_NAME_LEN - 1] = '\0';
					
					//add by lijing 20230110: 获取字段别名
					if(pColList->t[i].pName != NULL)
					{
						strncpy(tResult.columnOper[tResult.columnCount].AliasName, pColList->t[i].pName,  ZMDB_MAX_NAME_LEN-1); 
                    	tResult.columnOper[tResult.columnCount].AliasName[ZMDB_MAX_NAME_LEN - 1] = '\0'; 
					}
					
					tResult.columnOper[tResult.columnCount].bMin = true;
                    tResult.columnCount++;
					
					strncpy(pOrderBy[0].sParamName, pExpr->pList->t->pExpr->u.zToken, ZMDB_MAX_NAME_LEN - 1); 
					pOrderBy[0].sParamName[ZMDB_MAX_NAME_LEN - 1] = '\0';
					pOrderBy[0].bIsAsc = true; 
					OrderByCounts = 1;
					
					pLimit->iLeftLimitNum = 0;
					pLimit->iRightLimitNum = 1;
					
					tResult.tSelectType.bMin = true;
					tResult.tSelectType.bSort = true;					
					tResult.tSelectType.bLimit = true;
                    break; 
                case TK_SUM:
                    if(NULL == pExpr->pList 
                        || NULL == pExpr->pList->t
                        || NULL == pExpr->pList->t->pExpr
                        || NULL == pExpr->pList->t->pExpr->u.zToken
                        )
                    {
                        snprintf(szErr, sizeof(szErr),  "函数[sum]未指定列名."); 
                        sErrMsg = szErr; 
                        MDB_ERR("%s", sErrMsg.c_str()); 
                        return -1; 
                    }
					tResult.columnOper[tResult.columnCount].Clear();
                    strncpy(tResult.columnOper[tResult.columnCount].ColumnName, pExpr->pList->t->pExpr->u.zToken, ZMDB_MAX_NAME_LEN - 1); 
					tResult.columnOper[tResult.columnCount].ColumnName[ZMDB_MAX_NAME_LEN - 1] = '\0';
					
					//add by lijing 20230110: 获取字段别名
					if(pColList->t[i].pName != NULL)
					{
						strncpy(tResult.columnOper[tResult.columnCount].AliasName, pColList->t[i].pName,  ZMDB_MAX_NAME_LEN-1); 
                    	tResult.columnOper[tResult.columnCount].AliasName[ZMDB_MAX_NAME_LEN - 1] = '\0'; 
					}
					
					tResult.columnOper[tResult.columnCount].bSum = true;
                    tResult.columnCount++;
					tResult.tSelectType.bSum = true;
                    break; 
                case TK_TO_DATE:
                case TK_TO_CHAR:
                case TK_NVL: //虽然NVL在sql解析的时候是可以作为条件字段的，但是转换的时候限制作为查询字段
                    //指针在解析的时候已经判断
                    assert(pExpr->pList); 
                    pExpr = pExpr->pList->t[0].pExpr; 
                    assert(pExpr); 
                    if(TK_ID != pExpr->nOper)
                    {
                        snprintf(szErr, sizeof(szErr),  "第[%d]个查询字段格式不合法",  i+1); 
                        sErrMsg = szErr; 
                        MDB_ERR("%s", sErrMsg.c_str()); 
                        return -1; 
                    }
					tResult.columnOper[tResult.columnCount].Clear();
                    strncpy(tResult.columnOper[tResult.columnCount].ColumnName, pExpr->u.zToken,  ZMDB_MAX_NAME_LEN-1); 
                    tResult.columnOper[tResult.columnCount].ColumnName[ZMDB_MAX_NAME_LEN - 1] = '\0'; 

					//add by lijing 20230110: 获取字段别名
					if(pColList->t[i].pName != NULL)
					{
						strncpy(tResult.columnOper[tResult.columnCount].AliasName, pColList->t[i].pName,  ZMDB_MAX_NAME_LEN-1); 
                    	tResult.columnOper[tResult.columnCount].AliasName[ZMDB_MAX_NAME_LEN - 1] = '\0'; 
					}
					
                    tResult.columnCount++;
                    break; 
                case TK_ID:
					tResult.columnOper[tResult.columnCount].Clear();
                    strncpy(tResult.columnOper[tResult.columnCount].ColumnName, pExpr->u.zToken,  ZMDB_MAX_NAME_LEN-1); 
                    tResult.columnOper[tResult.columnCount].ColumnName[ZMDB_MAX_NAME_LEN - 1] = '\0'; 

					//add by lijing 20230110: 获取字段别名
					if(pColList->t[i].pName != NULL)
					{
						strncpy(tResult.columnOper[tResult.columnCount].AliasName, pColList->t[i].pName,  ZMDB_MAX_NAME_LEN-1); 
                    	tResult.columnOper[tResult.columnCount].AliasName[ZMDB_MAX_NAME_LEN - 1] = '\0'; 
					}
					
                    tResult.columnCount++;
                    break; 
                case TK_STRING:
                	//add by jyhk at 2023-03-13, reason:查询字段固定字符串,此时AliasName/ColumnName为字段名,ColumnName为字段值
					if (strlen(pExpr->u.zToken) >= ZMDB_MAX_NAME_LEN)
					{
						MDB_ERR("Select constant column'%s' over limit(31)!", pExpr->u.zToken); 
						return -1; 
					}

					tResult.columnOper[tResult.columnCount].Clear();
                    strncpy(tResult.columnOper[tResult.columnCount].ColumnName, pExpr->u.zToken,  ZMDB_MAX_NAME_LEN-1); 
                    tResult.columnOper[tResult.columnCount].ColumnName[ZMDB_MAX_NAME_LEN - 1] = '\0'; 
                    strncpy(tResult.columnOper[tResult.columnCount].ColumnValue, pExpr->u.zToken,  ZMDB_MAX_NAME_LEN-1); 
                    tResult.columnOper[tResult.columnCount].ColumnValue[ZMDB_MAX_NAME_LEN - 1] = '\0'; 
					tResult.columnOper[tResult.columnCount].bIsVirtual = true;

					//add by lijing 20230110: 获取字段别名
					if(pColList->t[i].pName != NULL)
					{
						strncpy(tResult.columnOper[tResult.columnCount].AliasName, pColList->t[i].pName,  ZMDB_MAX_NAME_LEN-1); 
                    	tResult.columnOper[tResult.columnCount].AliasName[ZMDB_MAX_NAME_LEN - 1] = '\0'; 
					}
					
                    tResult.columnCount++;
                    break; 
                default:
                    snprintf(szErr, sizeof(szErr),  "第[%d]个查询字段格式不支持",  i+1); 
                    sErrMsg = szErr; 
                    MDB_ERR("%s", sErrMsg.c_str()); 
                    return -1; 
                    break; 
            }//end switch oper
        }//end for

		/*add by jyhk at 2017-02-09, reason:特殊查询处理*/
		if (iLevel == 0)
		{			
			if(true == tResult.tSelectType.bDistinct)
			{
				if (tResult.columnCount > 4)
				{
					MDB_ERR("Select distinct have [%d] fields. Support only 4 fields!", tResult.columnCount); 
					return -1; 
				}
				OrderByCounts = tResult.columnCount;
				int i = 0;
				int j = 0;
				if (strcmp(tResult.columnOper[0].ColumnName, "count") == 0) 
				{
					i = 1;
					OrderByCounts--;
				}
				for (; i < tResult.columnCount; i++)
				{
					strncpy(pOrderBy[j].sParamName, tResult.columnOper[i].ColumnName, ZMDB_MAX_NAME_LEN - 1); 
					pOrderBy[j].sParamName[ZMDB_MAX_NAME_LEN - 1] = '\0'; 
					pOrderBy[j].bIsAsc = false; 
					j++;
				}
			}

		}

        return nRet; 
    }

    int OrderbyMap(TExprList *pExprList, TZmdbOrderBy* pOrderBy, int& OrderByCounts, string &sErrMsg)
    {
		if (OrderByCounts != 0)
		{//避免前面做了排序逻辑，后面又做一次排序
			return 0;
		}
		
        char szErr[256] = ""; 
        if(NULL  == pExprList) return 0; 
        assert(pOrderBy);
        assert(pExprList->t); 

        OrderByCounts = pExprList->nExpr; 
        if(OrderByCounts > ZMDB_MAX_SORT_COL_COUNTS || OrderByCounts <= 0)
        {
            sErrMsg =  "排序字段数不符合要求,[order by]目前最大只支持对[1~4]个字段进行排序"; 
            MDB_ERR("%s", sErrMsg.c_str()); 
            return -1; 
        }
		
        for(int i = 0;  i < OrderByCounts; ++i)
        {
            //排序字段
            TExpr *pOrderExpr = pExprList->t[i].pExpr; 
            assert(pOrderExpr); 
            if(TK_ID != pOrderExpr->nOper)
            {
                snprintf(szErr, sizeof(szErr), "第[%d]个排序字段格式不支持\n",  i+1); 
                sErrMsg = szErr; 
                MDB_ERR("%s", sErrMsg.c_str()); 
                return -1; 
            }
            assert(pOrderExpr->u.zToken); 
            strncpy(pOrderBy[i].sParamName, pOrderExpr->u.zToken, sizeof(pOrderBy[i].sParamName)); 
            pOrderBy[i].sParamName[sizeof(pOrderBy[i].sParamName) -1] = '\0'; 

            //排序顺序
            if (i == 0)
            {
                if((ORDER_ASC) == pExprList->t[i].nOrder)
                {
                    pOrderBy[i].bIsAsc = true; 
                }
                else
                {
                    pOrderBy[i].bIsAsc = false; 
                }
            }
            else
            {
                pOrderBy[i].bIsAsc = pOrderBy[0].bIsAsc; 
            }
        }

        return 0; 
    }

    int DistinctOrderbyMap(TExprList *pExprList, TZmdbOrderBy* pDistinct, int iDistinctCounts, string &sErrMsg)
    {
        char szErr[256] = ""; 
        bool bIsAsc = false;//add note by jyhk at 2017-06-20, reason: 现在多字段排序只支持统一升序或降序
        if(NULL  == pExprList) return 0; 
        assert(pDistinct);
        assert(pExprList->t); 

        int OrderByCounts = pExprList->nExpr; 
        if(OrderByCounts > ZMDB_MAX_SORT_COL_COUNTS || OrderByCounts <= 0)
        {
            sErrMsg =  "排序字段数不符合要求,[order by]目前最大只支持对[1~4]个字段进行排序"; 
            MDB_ERR("%s", sErrMsg.c_str()); 
            return -1; 
        }

		//检查排序字段是否都在distinct字段之中
        for(int i = 0;  i < OrderByCounts; ++i)
        {
            //排序字段
            TExpr *pOrderExpr = pExprList->t[i].pExpr; 
            assert(pOrderExpr); 
            if(TK_ID != pOrderExpr->nOper)
            {
                snprintf(szErr, sizeof(szErr),  "第[%d]个排序字段格式不支持\n",  i+1); 
                sErrMsg = szErr; 
                MDB_ERR("%s", sErrMsg.c_str()); 
                return -1; 
            }
            assert(pOrderExpr->u.zToken); 
            
			bool bFindFlag = false;
			int j = 0;
            for (; j < iDistinctCounts; j++)
            {
				if (0 == TZmdbStrFunc::StrCmpNoCase(pDistinct[j].sParamName, pOrderExpr->u.zToken, strlen(pDistinct[j].sParamName)))
				{
					bFindFlag = true;
					break;
				}
            }

            if (!bFindFlag)
            {
                snprintf(szErr, sizeof(szErr),  "排序字段[%s]不在distinct列表中,查询失败\n",  pOrderExpr->u.zToken); 
                sErrMsg = szErr; 
                MDB_ERR("%s", sErrMsg.c_str()); 
                return -1; 
            }

            //排序顺序
            if (i == 0)
            {
                if((ORDER_ASC) == pExprList->t[i].nOrder)
                {
                    bIsAsc = true; 
                }
                else
                {
                    bIsAsc = false; 
                }
            }

            pDistinct[j].bIsAsc = bIsAsc;
        }

		//根据orderby字段顺序重新分布排序字段顺序，因为distinct多字段时，不关心谁先谁后
        for(int i = 0;  i < OrderByCounts; ++i)
        {
        	TZmdbOrderBy tOrderbyTemp;

        	//调整顺序
			TExpr *pOrderExpr = pExprList->t[i].pExpr; 
			for (int j = 0; j < iDistinctCounts; j++)
			{
				if (0 == TZmdbStrFunc::StrCmpNoCase(pDistinct[j].sParamName, pOrderExpr->u.zToken, strlen(pDistinct[j].sParamName)))
				{//交换i、j位置
					memcpy(&tOrderbyTemp, &pDistinct[i], sizeof(TZmdbOrderBy));
					memcpy(&pDistinct[i], &pDistinct[j], sizeof(TZmdbOrderBy));
					memcpy(&pDistinct[j], &tOrderbyTemp, sizeof(TZmdbOrderBy));
					break;
				}
			}
		}
		
        return 0; 
    }

    /*add by jyhk at 2017-02-28, reason:分组支持*/
    int GroupbyMap(TExprList *pExprList, TZmdbGroupBy* pGroupBy, TZmdbOrderBy* pOrderBy, int& OrderByCounts, ZmdbParseSqlStruct &tResult, string &sErrMsg)
    {
        char szErr[256] = ""; 
        if(NULL  == pExprList) return 0; 
        assert(pGroupBy);
        assert(pExprList->t); 

		OrderByCounts += pExprList->nExpr;
		if (pExprList->nExpr != OrderByCounts)
		{//查询字段为MAX/MIN
			strncpy(pOrderBy[OrderByCounts - 1].sParamName, pOrderBy[0].sParamName, sizeof(pOrderBy[0].sParamName)); 
			pOrderBy[OrderByCounts - 1].sParamName[sizeof(pOrderBy[OrderByCounts - 1].sParamName) -1] = '\0'; 
			//调整MAX  MIN	的ASC顺序
			if(tResult.tSelectType.bMax)
			{
				pOrderBy[OrderByCounts - 1].bIsAsc = true;
			}
			else if (tResult.tSelectType.bMin)
			{
				pOrderBy[OrderByCounts - 1].bIsAsc = false;
			}
			//清理Limit
			tResult.tSelectType.bLimit = false;
		}
		
        if(OrderByCounts > ZMDB_MAX_SORT_COL_COUNTS || OrderByCounts <= 0)
        {
            sErrMsg =  "分组字段数不符合要求,[group by]目前只支持对[1~4]个字段进行分组;当使用MAX/MIN时,只支持对最大对3个字段进行分组"; 
            MDB_ERR("%s", sErrMsg.c_str()); 
            return -1; 
        }
		
        for(int i = 0;  i < pExprList->nExpr; ++i)
        {
            TExpr *pGroupExpr = pExprList->t[i].pExpr; 
            assert(pGroupExpr); 
            if(TK_ID != pGroupExpr->nOper)
            {
                snprintf(szErr, sizeof(szErr),  "第[%d]个分组字段格式不支持\n",  i+1); 
                sErrMsg = szErr; 
                MDB_ERR("%s", sErrMsg.c_str()); 
                return -1; 
            }
            assert(pGroupExpr->u.zToken); 

			//赋值排序字段
            strncpy(pOrderBy[i].sParamName, pGroupExpr->u.zToken, sizeof(pOrderBy[i].sParamName)); 
            pOrderBy[i].sParamName[sizeof(pOrderBy[i].sParamName) -1] = '\0'; 
            pOrderBy[i].bIsAsc = true;//默认升序
			
			//赋值分组字段
            strncpy(pGroupBy[i].sParamName, pGroupExpr->u.zToken, sizeof(pGroupBy[i].sParamName)); 
            pGroupBy[i].sParamName[sizeof(pGroupBy[i].sParamName) -1] = '\0'; 
			pGroupBy[i].iCounts = pExprList->nExpr;

			
        }

		//校验查询字段都为分组字段
        for(int i = 0;  i < tResult.columnCount; ++i)
        {
			if (tResult.columnOper[i].bCount || tResult.columnOper[i].bSum
				|| tResult.columnOper[i].bMax || tResult.columnOper[i].bMin) 
			{
				continue;
			}
			bool bFlag = false;
			for (int j = 0; j < OrderByCounts; j++)
			{
				if (0 == TZmdbStrFunc::StrCmpNoCase(tResult.columnOper[i].ColumnName, pOrderBy[j].sParamName))
				{
					bFlag = true;
					break;
				}
				bFlag = false;
			}
			if (false == bFlag)
			{
                snprintf(szErr, sizeof(szErr),  "查询字段[%s]不为分组字段!\n",  tResult.columnOper[i].ColumnName); 
                sErrMsg = szErr; 
                MDB_ERR("%s", sErrMsg.c_str()); 
                return -1; 
			}
        }

		
		tResult.tSelectType.bGroupby = true;
        return 0; 
    }

	
    /*modify by jyhk at 2016-11-20, reason:LIMIT支持*/
	int LimitMap(TExprList *pExprList, TZmdbLimit* pLimit, bool& bLimit, string &sErrMsg)
	{
		if (bLimit)
		{////避免前面做了分页逻辑max或min，后面又做一次分页
			return 0;
		}
		if(NULL  == pExprList) return 0; 
		assert(pLimit);
		assert(pExprList->t); 
	
		int iLimitCounts = pExprList->nExpr; 
		if(iLimitCounts == 1)
		{
			//当只有一个整型数据时，其形式就像0,X
			TExpr *pLimitExpr = pExprList->t[0].pExpr; 
			assert(pLimitExpr); 

			if (pLimitExpr->pLeft != NULL)
			{
				if (pLimitExpr->nOper == TK_UMINUS)
				{
					pLimit->iLeftLimitNum = 0;
					pLimit->iRightLimitNum = - pLimitExpr->pLeft->u.nValue;
				}
				else
				{
					pLimit->iLeftLimitNum = 0;
					pLimit->iRightLimitNum = pLimitExpr->pLeft->u.nValue;
				}
			}
			else
			{
				pLimit->iLeftLimitNum = 0;
				pLimit->iRightLimitNum = pLimitExpr->u.nValue;
			}
			bLimit = true;
		}
		else if (iLimitCounts == 2)
		{
			//当有二个整型数据时，其形式就像X,Y
			for(int i = 0;	i < iLimitCounts; ++i)
			{
				//排序字段
				TExpr *pLimitExpr = pExprList->t[i].pExpr; 
				assert(pLimitExpr); 
				if (0 == i)
				{
					if (pLimitExpr->pLeft != NULL)
					{
						if (pLimitExpr->nOper == TK_UMINUS)
						{
							pLimit->iLeftLimitNum = - pLimitExpr->pLeft->u.nValue;
							bLimit = true;
						}
						else
						{
							pLimit->iLeftLimitNum = pLimitExpr->pLeft->u.nValue;
							bLimit = true;
						}						
					}
					else
					{
						pLimit->iLeftLimitNum = pLimitExpr->u.nValue;
						bLimit = true;
					}
				}
				else
				{
					if (pLimitExpr->pLeft != NULL)
					{
						if (pLimitExpr->nOper == TK_UMINUS)
						{
							pLimit->iRightLimitNum = -pLimitExpr->pLeft->u.nValue;
							bLimit = true;
						}
						else
						{
							pLimit->iRightLimitNum = pLimitExpr->pLeft->u.nValue;
							bLimit = true;
						}						
					}
					else
					{
						pLimit->iRightLimitNum = pLimitExpr->u.nValue;
						bLimit = true;
					}

				}
			}
			bLimit = true;
		}
		else
		{
			sErrMsg =  "LIMIT字段数[%d]不符合要求,[LIMIT]目前只支持对[1]个或[2]个整型字段进行"; 
			MDB_ERR("%s", sErrMsg.c_str()); 
			return -1; 
		}

		
		if (pLimit->iRightLimitNum < -1)
		{
			sErrMsg =  "LIMIT操作数[%d]不符合要求,[LIMIT]查询条数必须不小于[-1]"; 
			MDB_ERR("%s", sErrMsg.c_str()); 
			return -1; 
		}

		
		if (pLimit->iLeftLimitNum < 0)
		{
			sErrMsg =  "LIMIT操作数[%d]不符合要求,[LIMIT]偏移量必须不小于[0]"; 
			MDB_ERR("%s", sErrMsg.c_str()); 
			return -1; 
		}

		if (pLimit->iRightLimitNum == -1)
		{
			bLimit = false;
		}
	
		return 0; 
	}

    int GenWhere(TExpr *pWhere, int &level, ZmdbParseSqlStruct &tResult, string &sErrMsg)
    {
        int nRet = 0; 
		int iTempOnClauseCnts = -1;
		int iTempWhereClauseCnts = -1;
        char szErr[256] = ""; 
        string sExpr; 

        if(NULL  == pWhere) return 0; 
        
        if (tResult.whereClauseCount >= (ZMDB_MAX_CONDITION_COUNT))
        {
            sErrMsg = "条件数超过限制"; 
            MDB_ERR("%s", sErrMsg.c_str()); 
            return -1; 
        }
        
        if(level > 1)
        {
            sErrMsg = "条件表达式目前只支持一层括号."; 
            MDB_ERR("%s", sErrMsg.c_str());
            return -1; 
        }
        if(level > 0)
        {
            tResult.totalLevel = level;  //即使in表达式只有一个比较值，也写成1
        }

        switch(pWhere->nOper)
        {
            case TK_AND:
                //填左操作数
                nRet = GenWhere(pWhere->pLeft, level, tResult, sErrMsg); 
                if(0 != nRet)
                {
                    MDB_CALL_ERR("GenWhere()"); 
                    return nRet; 
                }
                
                //赋值条件之间的关系
                if (tResult.whereClauseCount > 0)
                {
                	// 因为关联条件一定不会是二层条件, 说明此处的关系一定是上一个一层条件/二层条件的结束处, 与关联条件之间的关系
                	// 该关系如果为OR, 说明关联查询一层条件处有OR条件, 不支持直接报错
                	// 该关系如果为AND, 不需要做特殊处理
					if (tResult.whereClause[tResult.whereClauseCount-1].relation == 2) // 检测如下场景 ... OR A.TEST_ID = B.TEST_ID AND B.TEST_CHAR = '3'
					{
						ZMDB_TADD_ERROR("Join query only support 'and' relation!");
						throw TZmdbException("", "File=[%s], Line=[%d], Join query only support 'and' relation!", __FILE__, __LINE__);		
					}
					else
					{
						tResult.whereClause[tResult.whereClauseCount-1].relation = 1;  //1 --and
					}
				}
                
				//填右操作数
				nRet = GenWhere(pWhere->pRight, level, tResult, sErrMsg); 
				if(0 != nRet)
				{
					MDB_CALL_ERR("GenWhere()"); 
					return nRet; 
				}

                break; 
            case TK_OR: //与and一样
                //填左操作数
                nRet = GenWhere(pWhere->pLeft, level, tResult, sErrMsg);
                if(0 != nRet)
                {
                    MDB_CALL_ERR("GenWhere()"); 
                    return nRet; 
                }
                
                //赋值条件之间的关系
                if (tResult.whereClauseCount > 0)
                {
                	if (tResult.whereClause[tResult.whereClauseCount-1].relation != 0) // 检测如下场景 ... A.TEST_CHAR = '3' AND       A.TEST_ID = B.TEST_ID OR A.TEST_INT = 3 ...
            		{
            			// 可以确定上一个条件是关联条件, 关联条件一定不会是二层条件, 所以这里的条件一定是一层条件, 关联查询不允许一层条件是OR条件
						ZMDB_TADD_ERROR("Join query only support 'and' relation!");
						throw TZmdbException("", "File=[%s], Line=[%d], Join query only support 'and' relation!", __FILE__, __LINE__);		
            		}
					else
            		{
		                tResult.whereClause[tResult.whereClauseCount-1].relation = 2;  //2 --or
            		}
				}
				else // 检测如下场景       A.TEST_ID = B.TEST_ID OR A.TEST_INT = 3 ...
				{
					// 可以确定上一个条件是关联条件, 关联条件一定不会是二层条件, 所以这里的条件一定是一层条件, 关联查询不允许一层条件是OR条件
					ZMDB_TADD_ERROR("Join query only support 'and' relation!");
					throw TZmdbException("", "File=[%s], Line=[%d], Join query only support 'and' relation!", __FILE__, __LINE__);		
				}
				
                //填右操作数
                nRet = GenWhere(pWhere->pRight, level, tResult, sErrMsg); 
                if(0 != nRet)
                {
                    MDB_CALL_ERR("GenWhere()"); 
                    return nRet; 
                }
                break; 
            case TK_EQ:
            case TK_NE:
            case TK_GT:
            case TK_GE:
            case TK_LT:
            case TK_LE:
                nRet = FillColumn(pWhere, level, tResult, sErrMsg); 
                if(0 != nRet)
                {
                    MDB_CALL_ERR("FillColumn()"); 
                    return nRet; 
                }
                break; 
            case TK_IN:
                assert(pWhere->pLeft); 
                assert(pWhere->pList); 


                //填充字段类型判断
				if(TK_DOT == pWhere->pLeft->nOper)
				{
					/*add by jyhk at 2022-12-20, reason:表别名改造*/
					if (pWhere->pLeft->pLeft  == NULL || pWhere->pLeft->pLeft->nOper != TK_ID 
						|| pWhere->pLeft->pRight == NULL || pWhere->pLeft->pRight->nOper != TK_ID)
					{
						MDB_CALL_ERR("Where token used tk_dot'.', left/right oper must be id!"); 
						return -1; 
					}
					
					if (pWhere->pRight != NULL)
					{//关联条件
						MDB_CALL_ERR("In oper must use list condition in right oper!"); 
						return -1; 
					}
					else
					{
						tResult.whereClause[tResult.whereClauseCount].iUseMod = 0;
						strncpy(tResult.whereClause[tResult.whereClauseCount].TableAliasName, pWhere->pLeft->pLeft->u.zToken,  ZMDB_MAX_NAME_LEN-1); 
						tResult.whereClause[tResult.whereClauseCount].TableAliasName[ZMDB_MAX_NAME_LEN - 1] = '\0'; 
						strncpy(tResult.whereClause[tResult.whereClauseCount].ColumnName,  pWhere->pLeft->pRight->u.zToken, ZMDB_MAX_NAME_LEN) ; 
						tResult.whereClause[tResult.whereClauseCount].ColumnName[(ZMDB_MAX_NAME_LEN) - 1] = '\0'; 
					}
				}
				else if(TK_ID == pWhere->pLeft->nOper)
				{
					//填充字段
					strncpy(tResult.whereClause[tResult.whereClauseCount].ColumnName,  pWhere->pLeft->u.zToken, ZMDB_MAX_NAME_LEN) ; 
					tResult.whereClause[tResult.whereClauseCount].ColumnName[(ZMDB_MAX_NAME_LEN) - 1] = '\0'; 
				}
				else
                {
                    sErrMsg = "条件表达式中左操作数只支持字段名"; 
                    MDB_ERR("%s", sErrMsg.c_str());
                    return -1; 
                }
	
				//条件数判断
				if (pWhere->pList->nExpr > ZMDB_MAX_IN_CONDITION_COUNT)
				{
					sErrMsg = "IN条件数达到超过限制(1000)!"; 
					MDB_ERR("%s", sErrMsg.c_str()); 
					return -1; 
				}

				//填充比较符
                tResult.whereClause[tResult.whereClauseCount].Operator = _IN; 

				//填充条件
                for(int i = 0;  i < pWhere->pList->nExpr;  ++i)
                {
					sExpr = ""; 
                    TExpr *pExpr = pWhere->pList->t[i].pExpr; 

					/*add by jyhk at 2024-09-11, reason:支持IN(?)用法*/
					if (pExpr != NULL && pExpr->nOper == TK_WILDCARD)
					{
						// 已绑定变量的形式呈现
						int index = 0;
						for (int j = 0; j < tResult.whereClauseCount; j++) 
						{
							if (tResult.whereClause[j].bPlaceHolder)
							{
								index++;
							}
						}
						tResult.whereClause[tResult.whereClauseCount].bPlaceHolder = true;
						snprintf(tResult.whereClause[tResult.whereClauseCount].CompareParam, MAX_COLUMNVALUE_SIZE - 1, ":?%s_%d", tResult.whereClause[tResult.whereClauseCount].ColumnName, index) ; 
						tResult.whereClause[tResult.whereClauseCount].CompareParam[MAX_COLUMNVALUE_SIZE - 1] = '\0'; 
					}
					else
					{
						nRet = GenExpr(pExpr, sExpr, sErrMsg); 
						if(0 != nRet)
						{
							MDB_CALL_ERR("GenExpr()"); 
							return nRet; 
						}
						tResult.whereClause[tResult.whereClauseCount].setInParam.insert(sExpr);
					}
				}

				//填充当前层次
				tResult.whereClause[tResult.whereClauseCount].curLevel = 0;
				
				//累计条件数
				++tResult.whereClauseCount; 
                
#if 0
                //即使in表达式只有一个比较值，也写成1
                tResult.totalLevel = 1;  
                ++level;
                
                //填充第一个表达式开始标志
                tResult.whereClause[tResult.whereClauseCount].curTag = 1;    // 1 --开始
                
                for(int i = 0;  i < pWhere->pList->nExpr;  ++i)
                {

                    if (tResult.whereClauseCount >= (ZMDB_MAX_CONDITION_COUNT))
                    {
                        sErrMsg = "条件数超过限制"; 
                        MDB_ERR("%s", sErrMsg.c_str()); 
                        return -1; 
                    }
                    
                    if(i > 0)
                    {//设置前一个的关系为or
                        tResult.whereClause[tResult.whereClauseCount-1].relation = 2;  //1 --or
                    }
                    TExpr *pExpr = pWhere->pList->t[i].pExpr; 
                    //填充字段
                    strncpy(tResult.whereClause[tResult.whereClauseCount].ColumnName,  pWhere->pLeft->u.zToken, ZMDB_MAX_NAME_LEN) ; 
                    tResult.whereClause[tResult.whereClauseCount].ColumnName[(ZMDB_MAX_NAME_LEN) - 1] = '\0'; 
                    //填充比较符
                    tResult.whereClause[tResult.whereClauseCount].Operator = EQUALS; 
                    //填充条件
                    sExpr = ""; 
                    nRet = GenExpr(pExpr, sExpr, sErrMsg); 
                    if(0 != nRet)
                    {
                        MDB_CALL_ERR("GenExpr()"); 
                        return nRet; 
                    }
                    strncpy(tResult.whereClause[tResult.whereClauseCount].CompareParam,  sExpr.c_str(), MAX_COLUMNVALUE_SIZE - 1) ; 
                    tResult.whereClause[tResult.whereClauseCount].CompareParam[MAX_COLUMNVALUE_SIZE - 1] = '\0'; 
                    //填充当前层次
                    tResult.whereClause[tResult.whereClauseCount].curLevel = 1; //层次深一层
                    //累计条件数
                    ++tResult.whereClauseCount; 
                }
                --level;
                //填充最后一个表达式结束标志
                tResult.whereClause[tResult.whereClauseCount-1].curTag = 2;     // 2 --结束
                tResult.whereClause[tResult.whereClauseCount-1].curLevel = 0;   // 最后一个层次设置为当前层，而不是深一层
#endif
                break; 
            case TK_BRACKET:
            {
                //填充第一个表达式开始标志
				iTempOnClauseCnts = tResult.onClauseCount;
				iTempWhereClauseCnts = tResult.whereClauseCount;
                tResult.whereClause[tResult.whereClauseCount].curTag = 1;    // 1 --开始
                ++level;
                //填充表达式
                nRet = GenWhere(pWhere->pLeft, level, tResult, sErrMsg); 
                if(0 != nRet)
                {
                    MDB_CALL_ERR("GenWhere()"); 
                    return nRet; 
                }
                //填充最后一个表达式结束标志
                --level;

				// 赋值二层条件中间条件的 curTag=2（非首、非尾的比较行）
				// 原因：原实现循环内误写为 whereClause[count-1]，导致仅反复修改最后一行，中间行仍为 curTag=0。
				// 后果：括号内的 OR(relation==2) 落在 curTag=0 上，WhereMap 无法把 bHasOrOptionInLevel2 置真，
				// 在结束行(curTag=3)处被误判为「二层全 AND」并整段下放，丢失括号语义。
				// 方案：对下标区间 (首条+1)..(末条-1) 逐条设置 whereClause[i].curTag = 2。
				for (int i = iTempWhereClauseCnts + 1; i < tResult.whereClauseCount-1; i++)
				{
					tResult.whereClause[i].curTag = 2; // 2 -- 中间
				}

				// 检测二层条件中一定不能有关联条件
				if (iTempOnClauseCnts != tResult.onClauseCount)
				{
					ZMDB_TADD_ERROR("Join clause not support use '()'!");
					throw TZmdbException("", "File=[%s], Line=[%d], Join clause not support use '()'!", __FILE__, __LINE__);		
				}
				
                tResult.whereClause[tResult.whereClauseCount-1].curTag = 3;     // 3 --结束
                tResult.whereClause[tResult.whereClauseCount-1].curLevel = 0;   // 最后一个层次设置为当前层，而不是深一层
                break; 
            }
			case TK_NOT:
				//add by jyhk at 2023-02-06, reason:增加not in, not like支持
                assert(pWhere->pLeft); 

				if (pWhere->pLeft->nOper != TK_IN && pWhere->pLeft->nOper != TK_LIKE_KW)
				{
                    sErrMsg = "'NOT' expression only support 'in' and 'like'!"; 
                    MDB_ERR("%s", sErrMsg.c_str());
                    return -1; 
				}

            	//内部表达式填充
                nRet = GenWhere(pWhere->pLeft, level, tResult, sErrMsg);
                if(0 != nRet)
                {
                    MDB_CALL_ERR("GenWhere()"); 
                    return nRet; 
                }

                //填充表达式符号
				ConvertOperator(tResult.whereClause[tResult.whereClauseCount-1]);
                
				break;
			case TK_LIKE_KW:
				//add by jyhk at 2023-02-06, reason:增加like支持
                //填充字段类型判断
				if(TK_DOT == pWhere->pLeft->nOper)
				{
					/*add by jyhk at 2022-12-20, reason:表别名改造*/
					if (pWhere->pLeft->pLeft  == NULL || pWhere->pLeft->pLeft->nOper != TK_ID 
						|| pWhere->pLeft->pRight == NULL || pWhere->pLeft->pRight->nOper != TK_ID)
					{
						MDB_CALL_ERR("Where token used tk_dot'.', left/right oper must be id!"); 
						return -1; 
					}
					
					tResult.whereClause[tResult.whereClauseCount].iUseMod = 0;
					strncpy(tResult.whereClause[tResult.whereClauseCount].TableAliasName, pWhere->pLeft->pLeft->u.zToken,  ZMDB_MAX_NAME_LEN-1); 
					tResult.whereClause[tResult.whereClauseCount].TableAliasName[ZMDB_MAX_NAME_LEN - 1] = '\0'; 
					strncpy(tResult.whereClause[tResult.whereClauseCount].ColumnName,  pWhere->pLeft->pRight->u.zToken, ZMDB_MAX_NAME_LEN) ; 
					tResult.whereClause[tResult.whereClauseCount].ColumnName[(ZMDB_MAX_NAME_LEN) - 1] = '\0'; 
				}
				else if(TK_ID == pWhere->pLeft->nOper)
				{
					//填充字段
					strncpy(tResult.whereClause[tResult.whereClauseCount].ColumnName,  pWhere->pLeft->u.zToken, ZMDB_MAX_NAME_LEN) ; 
					tResult.whereClause[tResult.whereClauseCount].ColumnName[(ZMDB_MAX_NAME_LEN) - 1] = '\0'; 
				}
				else
                {
                    sErrMsg = "条件表达式中左操作数只支持字段名"; 
                    MDB_ERR("%s", sErrMsg.c_str());
                    return -1; 
                }

				//字符串校验
				if (pWhere->pRight->nOper != TK_STRING && pWhere->pRight->nOper != TK_COLON)
				{
					sErrMsg = "LIKE表达式只支持字符串!"; 
					MDB_ERR("%s", sErrMsg.c_str()); 
					return -1; 
				}
	
				//填充比较符
                tResult.whereClause[tResult.whereClauseCount].Operator = LIKE; 

				//填充条件
				sExpr = ""; 
				nRet = GenExpr(pWhere->pRight, sExpr, sErrMsg); 
				if(0 != nRet)
				{
					MDB_CALL_ERR("GenExpr()"); 
					return nRet; 
				}
				strncpy(tResult.whereClause[tResult.whereClauseCount].CompareParam,  sExpr.c_str(), MAX_COLUMNVALUE_SIZE - 1) ; 
				tResult.whereClause[tResult.whereClauseCount].CompareParam[MAX_COLUMNVALUE_SIZE - 1] = '\0'; 

				//填充当前层次
				tResult.whereClause[tResult.whereClauseCount].curLevel = 0;
				
				//累计条件数
				++tResult.whereClauseCount; 
				break;
            default:
                snprintf(szErr, sizeof(szErr),  "条件表达式格式[%d]尚不支持", pWhere->nOper); 
                sErrMsg = szErr; 
                MDB_ERR("%s", sErrMsg.c_str());
                return -1; 
                break; 
        }

        return nRet; 
    }

    int WhereMap(TExpr *pWhere, ZmdbParseSqlStruct &tResult,  string &sErrMsg)
    {//level / curtag /totallevel 
        int nRet = 0; 
        int level = 0; 
        if(NULL  == pWhere) return 0; 

        nRet = GenWhere(pWhere, level, tResult, sErrMsg); 
        if(0 != nRet)
        {
            MDB_CALL_ERR("GenExpr()"); 
            return nRet; 
        }

		// add by jyhk at 2024-11-19, reason: 二层条件拆解
		bool bHasOrOptionInLevel1 = false;
		bool bHasOrOptionInLevel2 = false;
		int iStartPos = -1;
		for (int i = 0; i < tResult.whereClauseCount; i++)
		{
			if (tResult.whereClause[i].curTag == 0)
			{
				// 记录一层条件
				if (tResult.whereClause[i].relation == 2)
				{
					bHasOrOptionInLevel1  = true;
				}
			}
			else if (tResult.whereClause[i].curTag == 1)
			{
				// 二层开始阶段
				iStartPos = i;
				bHasOrOptionInLevel2 = false;
				if (tResult.whereClause[i].relation == 2)
				{
					bHasOrOptionInLevel2  = true;
				}
			}
			else if (tResult.whereClause[i].curTag == 2)
			{
				// 二层中间阶段
				if (tResult.whereClause[i].relation == 2)
				{
					bHasOrOptionInLevel2  = true;
				}
			}
			else
			{
				// 二层结束阶段（curTag==3，或其它非 0/1/2 的结束标记）
				// 原因：若 GenWhere 未给括号内中间行打上 curTag=2，则括号内的 OR 只会在 curTag=0 分支被记入
				// bHasOrOptionInLevel1，bHasOrOptionInLevel2 仍为 false，会错误执行「全 AND 下放」。
				// 方案：在准备下放前，扫描本括号块 [iStartPos, i] 内，若存在任一行 j 与下一行以 OR 连接
				// (relation==2)，则视为二层含 OR，禁止下放；并要求 iStartPos 有效，避免未匹配到块首时误写。
				if (bHasOrOptionInLevel2 == false && iStartPos >= 0)
				{
					bool bHasOrInsideBracket = false;
					for (int j = iStartPos; j < i; j++)
					{
						if (tResult.whereClause[j].relation == 2)
						{
							bHasOrInsideBracket = true;
							break;
						}
					}
					if (!bHasOrInsideBracket)
					{
						// 如果二层条件全部是AND, 就下放到一层条件
						for (int j = iStartPos; j <= i; j++)
						{
							tResult.whereClause[j].curLevel = 0;
							tResult.whereClause[j].curTag = 0;
						}
					}
				}
			}
		}

		// add by jyhk at 2024-11-19, reason: 关联查询一层条件不允许是OR条件
    	if (tResult.fromClauseCount > 1) // 检测如下场景 ... A.TEST_INT = 3 OR A.TEST_ID = B.TEST_ID ...
    	{
			//检查所有的条件都必须是AND
			if (bHasOrOptionInLevel1)
			{
				ZMDB_TADD_ERROR("Join query only support 'and' relation!");
				throw TZmdbException("", "File=[%s], Line=[%d], Join query only support 'and' relation!", __FILE__, __LINE__);		
			}
    	}

        return nRet; 
    }

	int CheckOnCluase(TExpr *pWhere, string &sErrMsg)
	{
		// 条件树的遍历递归实现是从右往左, 因此pWhere描述的第一个条件实际上是SQL中最后一个条件
		// 因此, pWhere的左操作数一般代表了递归, pWhere的右操作数一般代表了赋值
		// 具体打个比方:select * from test01 where test_id = 1 and test_id = 2 and a.test_id = 3;
		// 此时pWhere及其pLeft节点都是从右至左落在and这个操作上的, 除非到了第一个and
		if(NULL == pWhere)
		{
			return 0; 
		}

		switch(pWhere->nOper)
		{
			case TK_AND:
			{
				// 发现是AND就继续递归检查
				int nRet = CheckOnCluase(pWhere->pLeft, sErrMsg); 
				if(0 != nRet)
				{
					MDB_CALL_ERR("CheckOnCluase()"); 
					return nRet; 
				}
				break;
			}
			case TK_OR:
			{
				// 发现是OR就报错
				sErrMsg = "On clause not support 'or' option!"; 
				MDB_ERR("%s", sErrMsg.c_str());
				return -1;
			}
			case TK_BRACKET:
			{
				// 发现是子条件也报错
				sErrMsg = "On clause not support sub clause like '()'!"; 
				MDB_ERR("%s", sErrMsg.c_str());
				return -1; 
			}
			default:
				break; 
		}

		return 0; 

	}

	// on操作暂时不支持带子条件或OR条件
    int OnMap(TSrcList*pSrcList, ZmdbParseSqlStruct &tResult,  string &sErrMsg)
    {
        int nRet = 0; 

		// add by jyhk at 2024-11-19, reason: on操作暂时不支持带子条件或OR条件
		for (int i = 1; i < pSrcList->nId; ++i)
		{
			int level = 0;
			nRet = CheckOnCluase(pSrcList->t[i].pJoinOn, sErrMsg);
			if(0 != nRet)
			{
				MDB_CALL_ERR("CheckOnCluase()"); 
				return nRet; 
			}
		}

        // 进行on操作到onClause上的映射
		for (int i = 1; i < pSrcList->nId; ++i)
		{
			int level = 0;
			nRet = GenWhere(pSrcList->t[i].pJoinOn, level, tResult, sErrMsg); 
			if(0 != nRet)
			{
				MDB_CALL_ERR("GenExpr()"); 
				return nRet; 
			}
		}
		
        return nRet; 

    }

					break;
				}
				case TK_INTEGER:
				{
					//取模值
					tResult.whereClause[tResult.whereClauseCount].llModValue = pExpr->u.nValue;
					break;
				}
				default:
				{
					snprintf(szErr, sizeof(szErr),	"第[%d]个取模字段格式不支持",  i+1); 
					sErrMsg = szErr; 
					MDB_ERR("%s", sErrMsg.c_str()); 
					return -1; 
				}
			}
		}
		return iRet;
	}

	int FillFuncColumn(int iFuncType, TExprList *pFuncList, ZmdbParseSqlStruct &tResult, string &sErrMsg)
	{
		int iRet = 0;
		assert(pFuncList);
		assert(pFuncList->t);

		switch(iFuncType)
		{
			case FUNC_ADD_MONTH:
			{
				if (2 != pFuncList->nExpr)
				{
					sErrMsg = "ADD_MONTH右操作数有且只有2个表达式"; 
					MDB_ERR("%s", sErrMsg.c_str());
					return -1; 
				}

				char sTime[MAX_TIME_LEN] = {0};
				long long llOffSet = 0;
				bool bMinus = false;
				//需要操作的日期
				TExpr *pExprDate = pFuncList->t[0].pExpr;
				if (TK_STRING == pExprDate->nOper)
				{
					if (14 == strlen(pExprDate->u.zToken))
					{
						strncpy(sTime, pExprDate->u.zToken, MAX_TIME_LEN - 1);
						sTime[MAX_TIME_LEN - 1] = '\0';
					}
					else
					{
						sErrMsg = "ADD_MONTH右操作数日期只支持YYYYMMDDhhmmss这种形式"; 
						MDB_ERR("%s", sErrMsg.c_str());
						return -1; 
					}
				}
				else if (TK_SYSDATE == pExprDate->nOper)
				{
					TZmdbDateTimeFunc::GetCurrentTimeStr(sTime);
				}
				else
				{
					sErrMsg = "ADD_MONTH右操作数日期只支持14位字符串(YYYYMMDDhhmmss)或SYSDATE"; 
					MDB_ERR("%s", sErrMsg.c_str());
					return -1; 
				}
				
				TExpr *pOffSetDate = pFuncList->t[1].pExpr;
				//需要偏移的月份
				if (TK_INTEGER == pOffSetDate->nOper)
				{
					llOffSet = pOffSetDate->u.nValue;
				}
				else if (TK_UMINUS == pOffSetDate->nOper)
				{
					bMinus = true;
					llOffSet = pOffSetDate->pLeft->u.nValue;
				}
				else
				{
					sErrMsg = "ADD_MONTH右操作数偏移只支持整型"; 
					MDB_ERR("%s", sErrMsg.c_str());
					return -1; 
				}

				TZmdbDateTimeFunc::AddMonths(sTime, llOffSet, bMinus);

				//最后赋值结果
				strncpy(tResult.whereClause[tResult.whereClauseCount].CompareParam, sTime, MAX_COLUMNVALUE_SIZE);
				break;
			}
			case FUNC_DATE_ADD:
			{
				if (3 != pFuncList->nExpr)
				{
					sErrMsg = "DATE_ADD右操作数有且只有3个表达式"; 
					MDB_ERR("%s", sErrMsg.c_str());
					return -1; 
				}

				char sTime[MAX_TIME_LEN] = {0};
				
				//需要操作的日期
				TExpr *pExprDate = pFuncList->t[0].pExpr;
				if (TK_STRING == pExprDate->nOper)
				{
					if (14 == strlen(pExprDate->u.zToken))
					{
						strncpy(sTime, pExprDate->u.zToken, MAX_TIME_LEN - 1);
						sTime[MAX_TIME_LEN - 1] = '\0';
					}
					else
					{
						sErrMsg = "DATE_ADD右操作数日期只支持YYYYMMDDhhmmss这种形式字符串"; 
						MDB_ERR("%s", sErrMsg.c_str());
						return -1; 
					}
					strncpy(tResult.whereClause[tResult.whereClauseCount].CompareParam, sTime, MAX_COLUMNVALUE_SIZE - 1);
					tResult.whereClause[tResult.whereClauseCount].CompareParam[MAX_COLUMNVALUE_SIZE - 1] = '\0'; 
				}
				else if (TK_SYSDATE == pExprDate->nOper || TK_COLON == pExprDate->nOper)
				{
					//填充条件
			        string sExpr; 
					iRet = GenExpr(pExprDate, sExpr, sErrMsg); 
					if(0 != iRet)
					{
						MDB_CALL_ERR("GenExpr()"); 
						return iRet; 
					}
					strncpy(tResult.whereClause[tResult.whereClauseCount].CompareParam,  sExpr.c_str(), MAX_COLUMNVALUE_SIZE - 1) ; 
					tResult.whereClause[tResult.whereClauseCount].CompareParam[MAX_COLUMNVALUE_SIZE - 1] = '\0'; 
				}
				else
				{
					sErrMsg = "DATE_ADD右操作数日期只支持14位字符串(YYYYMMDDhhmmss)或SYSDATE或者绑定变量"; 
					MDB_ERR("%s", sErrMsg.c_str());
					return -1; 
				}

				
				//需要偏移的时间
				TExpr *pOffSetDate = pFuncList->t[1].pExpr;
				if (TK_INTEGER == pOffSetDate->nOper)
				{
					tResult.whereClause[tResult.whereClauseCount].llTimeInterval = pOffSetDate->u.nValue;
				}
				else if (TK_UMINUS == pOffSetDate->nOper)
				{
					tResult.whereClause[tResult.whereClauseCount].bMinus = true;
					tResult.whereClause[tResult.whereClauseCount].llTimeInterval = pOffSetDate->pLeft->u.nValue;
				}
				else
				{
					sErrMsg = "DATE_ADD右操作数偏移只支持整型"; 
					MDB_ERR("%s", sErrMsg.c_str());
					return -1; 
				}

				//需要偏移的时间单位
				TExpr *pOffSetDateUnit = pFuncList->t[2].pExpr;
				if (TK_UNIT == pOffSetDateUnit->nOper)
				{
					tResult.whereClause[tResult.whereClauseCount].duUnit = (DATE_UNIT)pOffSetDateUnit->u.nValue;
				}
				else
				{
					sErrMsg = "DATE_ADD右操作数单位只支持YEAR|MONTH|DAY|HOUR|MINUTE|SECOND"; 
					MDB_ERR("%s", sErrMsg.c_str());
					return -1; 
				}
				
				break;
			}
		}
		return iRet;
	}
	
    int FillColumn(TExpr *pWhere, int level, ZmdbParseSqlStruct &tResult, string &sErrMsg)
    {//不支持多重表达式嵌套
        int nRet = 0; 
        assert(pWhere); 
        assert(pWhere->pLeft); 
        assert(pWhere->pRight); 

        //填充比较符
        switch(pWhere->nOper)
        {
            case TK_EQ:
                tResult.whereClause[tResult.whereClauseCount].Operator = EQUALS; 
                break; 
            case TK_NE:
                tResult.whereClause[tResult.whereClauseCount].Operator = NOT_EQUALS; 
                break; 
            case TK_GT:
                tResult.whereClause[tResult.whereClauseCount].Operator = GREATER_THAN; 
                break; 
            case TK_GE:
                tResult.whereClause[tResult.whereClauseCount].Operator = GREATER_EQUALS_THAN; 
                break; 
            case TK_LT:
                tResult.whereClause[tResult.whereClauseCount].Operator = LESS_THAN; 
                break; 
            case TK_LE:
                tResult.whereClause[tResult.whereClauseCount].Operator = LESS_EQUALS_THAN; 
                break; 
            default:
                assert(0); 
                break; 
        }

        //填充字段
        if(TK_ID != pWhere->pLeft->nOper && TK_MOD != pWhere->pLeft->nOper && TK_DOT != pWhere->pLeft->nOper)
        {
            sErrMsg = "条件表达式左操作数只支持字段名、表别名及取模函数"; 
            MDB_ERR("%s", sErrMsg.c_str());
            return -1; 
        }

		/*add by ji.yanghongkang at 2016-05-26, reason:如果关键字是MOD，就替换条件表达式的左操作数*/
		if (TK_MOD == pWhere->pLeft->nOper)
		{
			tResult.whereClause[tResult.whereClauseCount].iUseMod = 1;
			nRet = FillMODColumn(pWhere->pLeft->pList, tResult, sErrMsg);
			//还需要检测该字段类型是否是整型，放到外面校验
		}
		else if (TK_ID == pWhere->pLeft->nOper)
		{
			if (TK_ADD_MONTH == pWhere->pRight->nOper)
			{
				//填充字段
				strncpy(tResult.whereClause[tResult.whereClauseCount].ColumnName,  pWhere->pLeft->u.zToken, ZMDB_MAX_NAME_LEN) ; 
				tResult.whereClause[tResult.whereClauseCount].ColumnName[(ZMDB_MAX_NAME_LEN) - 1] = '\0'; 
			
				//填充条件值
				tResult.whereClause[tResult.whereClauseCount].iFunctionFlag = FUNC_ADD_MONTH;
				nRet = FillFuncColumn(FUNC_ADD_MONTH, pWhere->pRight->pList, tResult, sErrMsg);
				
				//填充当前层次
				tResult.whereClause[tResult.whereClauseCount].curLevel = level; 
				
				//累计条件数
				++tResult.whereClauseCount; 
			
				return nRet;
			}
			else if (TK_DATE_ADD == pWhere->pRight->nOper)
			{
				//填充字段
				strncpy(tResult.whereClause[tResult.whereClauseCount].ColumnName,  pWhere->pLeft->u.zToken, ZMDB_MAX_NAME_LEN) ; 
				tResult.whereClause[tResult.whereClauseCount].ColumnName[(ZMDB_MAX_NAME_LEN) - 1] = '\0'; 
			
				//填充条件值
				tResult.whereClause[tResult.whereClauseCount].iFunctionFlag = FUNC_DATE_ADD;
				nRet = FillFuncColumn(FUNC_DATE_ADD, pWhere->pRight->pList, tResult, sErrMsg);
								
				//填充当前层次
				tResult.whereClause[tResult.whereClauseCount].curLevel = level; 
				
				//累计条件数
				++tResult.whereClauseCount; 
			
				return nRet;
			}
			else
			{
				tResult.whereClause[tResult.whereClauseCount].iUseMod = 0;
				strncpy(tResult.whereClause[tResult.whereClauseCount].ColumnName,  pWhere->pLeft->u.zToken, ZMDB_MAX_NAME_LEN) ; 
				tResult.whereClause[tResult.whereClauseCount].ColumnName[(ZMDB_MAX_NAME_LEN) - 1] = '\0'; 
			}			
		}
		else if (TK_DOT == pWhere->pLeft->nOper)
		{
			/*add by jyhk at 2022-12-20, reason:表别名改造*/
			if (pWhere->pLeft->pLeft  == NULL || pWhere->pLeft->pLeft->nOper != TK_ID 
				|| pWhere->pLeft->pRight == NULL || pWhere->pLeft->pRight->nOper != TK_ID)
			{
				MDB_CALL_ERR("Where token used tk_dot'.', left/right oper must be id!"); 
				return -1; 
			}

			//此处用别名相同来判断是否同一张表
			if (TK_DOT == pWhere->pRight->nOper)
			{//关联条件
				if (pWhere->pRight->pLeft  == NULL || pWhere->pRight->pLeft->nOper != TK_ID 
					|| pWhere->pRight->pRight == NULL || pWhere->pRight->pRight->nOper != TK_ID)
				{
					MDB_CALL_ERR("Where token used tk_dot'.', left/right oper must be id!"); 
					return -1; 
				}

				// add note by jyhk at 2024-10-15: 右操作数的操作符是点的情况, 只有关联条件, 及字段比较
				if (strcmp(pWhere->pRight->pLeft->u.zToken,pWhere->pLeft->pLeft->u.zToken) == 0)
				{
					// 一定是字段比较
					tResult.whereClause[tResult.whereClauseCount].iUseMod = 0;
					strncpy(tResult.whereClause[tResult.whereClauseCount].TableAliasName, pWhere->pLeft->pLeft->u.zToken,  ZMDB_MAX_NAME_LEN-1); 
					tResult.whereClause[tResult.whereClauseCount].TableAliasName[ZMDB_MAX_NAME_LEN - 1] = '\0'; 
					strncpy(tResult.whereClause[tResult.whereClauseCount].ColumnName, pWhere->pLeft->pRight->u.zToken, ZMDB_MAX_NAME_LEN) ; 
					tResult.whereClause[tResult.whereClauseCount].ColumnName[(ZMDB_MAX_NAME_LEN) - 1] = '\0'; 

					tResult.whereClause[tResult.whereClauseCount].bCompareColumn = true;
					strncpy(tResult.whereClause[tResult.whereClauseCount].CompareParam, pWhere->pRight->pRight->u.zToken, MAX_COLUMNVALUE_SIZE - 1) ; 
					tResult.whereClause[tResult.whereClauseCount].CompareParam[MAX_COLUMNVALUE_SIZE - 1] = '\0'; 
					
					//填充当前层次
					tResult.whereClause[tResult.whereClauseCount].curLevel = level; 
					
					//累计条件数
					++tResult.whereClauseCount; 

					return nRet;
				}
				else
				{
					// 一定是关联条件
					strncpy(tResult.onClause[tResult.onClauseCount].LeftTableAliasName, pWhere->pLeft->pLeft->u.zToken,  ZMDB_MAX_NAME_LEN-1); 
					tResult.onClause[tResult.onClauseCount].LeftTableAliasName[ZMDB_MAX_NAME_LEN - 1] = '\0'; 
					strncpy(tResult.onClause[tResult.onClauseCount].LeftColumnName,  pWhere->pLeft->pRight->u.zToken, ZMDB_MAX_NAME_LEN) ; 
					tResult.onClause[tResult.onClauseCount].LeftColumnName[(ZMDB_MAX_NAME_LEN) - 1] = '\0'; 
					strncpy(tResult.onClause[tResult.onClauseCount].RightTableAliasName, pWhere->pRight->pLeft->u.zToken,  ZMDB_MAX_NAME_LEN-1); 
					tResult.onClause[tResult.onClauseCount].RightTableAliasName[ZMDB_MAX_NAME_LEN - 1] = '\0'; 
					strncpy(tResult.onClause[tResult.onClauseCount].RightColumnName,  pWhere->pRight->pRight->u.zToken, ZMDB_MAX_NAME_LEN) ; 
					tResult.onClause[tResult.onClauseCount].RightColumnName[(ZMDB_MAX_NAME_LEN) - 1] = '\0'; 
					tResult.onClauseCount++;
					return nRet; 
				}	
			}
			else
			{
				tResult.whereClause[tResult.whereClauseCount].iUseMod = 0;
				strncpy(tResult.whereClause[tResult.whereClauseCount].TableAliasName, pWhere->pLeft->pLeft->u.zToken,  ZMDB_MAX_NAME_LEN-1); 
				tResult.whereClause[tResult.whereClauseCount].TableAliasName[ZMDB_MAX_NAME_LEN - 1] = '\0'; 
				strncpy(tResult.whereClause[tResult.whereClauseCount].ColumnName,  pWhere->pLeft->pRight->u.zToken, ZMDB_MAX_NAME_LEN) ; 
				tResult.whereClause[tResult.whereClauseCount].ColumnName[(ZMDB_MAX_NAME_LEN) - 1] = '\0'; 
			}
		}
		
		//填充条件
        string sExpr; 
		nRet = GenExpr(pWhere->pRight, sExpr, sErrMsg); 
		if(0 != nRet)
		{
			MDB_CALL_ERR("GenExpr()"); 
			return nRet; 
		}

		// add by jyhk at 2024-09-20, reason:字段比较判断, 避免放入分片字段中导致分片哈希计算出错
		if (pWhere->pRight->nOper == TK_ID)
		{
			tResult.whereClause[tResult.whereClauseCount].bCompareColumn = true;
		}

		strncpy(tResult.whereClause[tResult.whereClauseCount].CompareParam,  sExpr.c_str(), MAX_COLUMNVALUE_SIZE - 1) ; 
		tResult.whereClause[tResult.whereClauseCount].CompareParam[MAX_COLUMNVALUE_SIZE - 1] = '\0'; 

        //填充当前层次
        tResult.whereClause[tResult.whereClauseCount].curLevel = level; 

        //累计条件数
        ++tResult.whereClauseCount; 

        return nRet; 
    }

    int InsertColumnMap(TInsert *pInsert, ZmdbParseSqlStruct &tResult, string &sErrMsg)
    {
        int nRet = 0; 
        assert(pInsert); 
        TIdList   *pCol = pInsert->pCol;
        assert(pCol); 
        for(int i = 0;  i < pCol->nId;  ++i)
        {
            char *pName = pCol->t[i].pName; 
            assert(pName); 

            if(tResult.columnCount  >=  (MAX_COLUMN_COUNT))
            {
                sErrMsg = "插入字段数超过限制!"; 
                MDB_ERR("%s", sErrMsg.c_str());
                return -1;
            }

            strncpy(tResult.columnOper[tResult.columnCount].ColumnName, pName, ZMDB_MAX_NAME_LEN); 
            tResult.columnOper[tResult.columnCount].ColumnName[(ZMDB_MAX_NAME_LEN) - 1] = '\0'; 
            tResult.columnCount++;
        }//end for

        TExprList *pValue = pInsert->pValue; 
        for(int i = 0;  i < pValue->nExpr;  ++i)
        {
            string sExpr; 
            nRet = GenExpr(pValue->t[i].pExpr, sExpr, sErrMsg); 
            if(0 != nRet)
            {
                MDB_CALL_ERR("GenExpr()"); 
                return nRet; 
            }
            strncpy(tResult.columnOper[i].ColumnValue, sExpr.c_str(), MAX_COLUMNVALUE_SIZE); 
            tResult.columnOper[i].ColumnValue[(MAX_COLUMNVALUE_SIZE) - 1] = '\0'; 
        }
        //不需要验证数量是否一致，解析的时候已验证过

        return nRet; 
    }

    ///< update 和 select 的列表达式不同的是 ，update有eq字段
    int UpdateColumnMap(TExprList *pColList, ZmdbParseSqlStruct &tResult, string &sErrMsg)
    {
        int nRet = 0; 
        assert(pColList); 

        // mdy by jyhk at 2024-06-05, reason: update 字段支持别名
        for(int i = 0;  i < pColList->nExpr;  ++i)
        {
        	// 更新字段超量检测
            if(tResult.columnCount  >=  (MAX_COLUMN_COUNT))
            {
                sErrMsg = "更新字段数超过限制!"; 
                MDB_ERR("%s", sErrMsg.c_str());
                return -1;
            }
        
			// 获取当前字段表达式
			TExpr *pExpr = pColList->t[i].pExpr;
			if (pExpr == NULL)
			{
                sErrMsg = "更新字段表达式为空, 请检查SQL是否符合规范!"; 
                MDB_ERR("%s", sErrMsg.c_str());
                return -1;
			}

			// 更新字段必须是等号表达式
			if (pExpr->nOper != TK_EQ)
			{
                sErrMsg = "更新字段只支持'='表达式!"; 
                MDB_ERR("%s", sErrMsg.c_str());
                return -1;
			}

			// 表达式左值test_id/a.test_id
			TExpr *pLeftExpr = pExpr->pLeft;
			// 表达式右值a.test_id + 1 / 1
			TExpr *pRightExpr = pExpr->pRight;

			// 表达式左右值存在检测
			if (pLeftExpr == NULL || pRightExpr == NULL)
			{
                sErrMsg = "更新字段左/右值为空, 请检查SQL是否符合规范!"; 
                MDB_ERR("%s", sErrMsg.c_str());
                return -1;
			}

			// 解析表达式左值 -- 赋值操作字段
			if (pLeftExpr->nOper == TK_DOT)
			{
				TExpr *pAliasExpr = pLeftExpr->pLeft;
				TExpr *pColumnExpr = pLeftExpr->pRight;
                assert(pAliasExpr); 
                assert(pColumnExpr); 
                
				// add note by jyhk at 2024-06-06: 目前别名只做稽核
				if (TK_ID != pAliasExpr->nOper || TZmdbStrFunc::StrCmpNoCase(pAliasExpr->u.zToken, tResult.tableAliasName) != 0)
				{
					sErrMsg = "更新字段中表别名与操作表不匹配"; 
					MDB_ERR("%s", sErrMsg.c_str());
					return -1;
				}

				//字段
				strncpy(tResult.columnOper[tResult.columnCount].ColumnName, pColumnExpr->u.zToken, ZMDB_MAX_NAME_LEN); 
				tResult.columnOper[tResult.columnCount].ColumnName[(ZMDB_MAX_NAME_LEN) - 1] = '\0'; 
			}
			else
			{
				//字段
				strncpy(tResult.columnOper[tResult.columnCount].ColumnName, pLeftExpr->u.zToken, ZMDB_MAX_NAME_LEN); 
				tResult.columnOper[tResult.columnCount].ColumnName[(ZMDB_MAX_NAME_LEN) - 1] = '\0'; 
			}

			// 解析表达式右值 -- 稽核操作字段, 取出操作值
            if(TK_PLUS == pRightExpr->nOper || TK_MINUS == pRightExpr->nOper)
            {
            	// 解析增量表达式中左值 -- 表别名判断
                TExpr * pRightLeftExpr = pRightExpr->pLeft;
                assert(pRightLeftExpr); 

				// 解析别名
				if (pRightLeftExpr->nOper == TK_DOT)
				{
					TExpr *pAliasExpr = pRightLeftExpr->pLeft;
					TExpr *pColumnExpr = pRightLeftExpr->pRight;
					assert(pAliasExpr); 
					assert(pColumnExpr); 
					
					// add note by jyhk at 2024-06-06: 目前别名只做稽核
					if (TK_ID != pAliasExpr->nOper || TZmdbStrFunc::StrCmpNoCase(pAliasExpr->u.zToken, tResult.tableAliasName) != 0)
					{
						sErrMsg = "更新字段中表别名与操作表不匹配"; 
						MDB_ERR("%s", sErrMsg.c_str());
						return -1;
					}
				
					// 相同字段稽核
					if(TK_ID != pColumnExpr->nOper || 0 != TZmdbStrFunc::StrNoCaseCmp(pColumnExpr->u.zToken, tResult.columnOper[tResult.columnCount].ColumnName))
					{
						sErrMsg = "增量更新字段只支持相同字段操作"; 
						MDB_ERR("%s", sErrMsg.c_str()); 
						return -1; 
					}
				}
				else
				{
					// 相同字段稽核
					if(TK_ID != pRightLeftExpr->nOper || 0 != TZmdbStrFunc::StrNoCaseCmp(pRightLeftExpr->u.zToken, tResult.columnOper[tResult.columnCount].ColumnName))
					{
						sErrMsg = "增量更新字段只支持相同字段操作"; 
						MDB_ERR("%s", sErrMsg.c_str()); 
						return -1; 
					}					
				}

            }

			// 生成值表达式
			string sExpr; 
			nRet = GenExpr(pRightExpr, sExpr, sErrMsg); 
			if(0 != nRet)
			{
				MDB_CALL_ERR("GenExpr()"); 
				return nRet; 
			}
			strncpy(tResult.columnOper[tResult.columnCount].ColumnValue, sExpr.c_str(), MAX_COLUMNVALUE_SIZE); 
			tResult.columnOper[tResult.columnCount].ColumnValue[(MAX_COLUMNVALUE_SIZE) - 1] = '\0'; 

            //列计数
            ++tResult.columnCount; 
        }//end for

        return nRet; 
    }

	int SelectOptionMap(TOption * pOption, ZmdbParseSqlStruct &tResult)
	{
		if (NULL == pOption)
		{
			return 0;
		}

		tResult.tSelectOption.bForUpdate = pOption->bForUpdate;
		tResult.tSelectOption.bNoWait = pOption->bNoWait;
		tResult.tSelectOption.iForUpdateWaitTime = pOption->iForUpdateWaitTime;
		tResult.tSelectType.bForupdate = pOption->bForUpdate;
		tResult.tSelectType.bDistinct = pOption->bDistinct;
		return 0;
	}

	int SelectFromMap(TSrcList* pSrcList, ZmdbParseSqlStruct &tResult,  string &sErrMsg)
	{
		/*mdy by jyhk at 2022-12-19, reason:关联查询多表改造*/
		tResult.fromClauseCount = 0;
		for (int i = 0; i < pSrcList->nId; i++)
		{
			strncpy(tResult.fromClause[tResult.fromClauseCount].TableName, 
				pSrcList->t[i].pTable->szTabName, ZMDB_MAX_NAME_LEN);
			if (pSrcList->t[i].pTable->szTabAlias == NULL)
			{
				if ( pSrcList->nId > 1)
				{
					sErrMsg = "关联查询必须使用表别名";
					MDB_ERR("Join query must use alias table name!"); 
					return -1;
				}
			}
			else
			{
				strncpy(tResult.fromClause[tResult.fromClauseCount].AliasName, 
					pSrcList->t[i].pTable->szTabAlias, ZMDB_MAX_NAME_LEN);
			}
			tResult.fromClause[tResult.fromClauseCount].jointype = (JOIN_TYPE)pSrcList->t[i].jointype;
			tResult.fromClauseCount++;
		}

		if (tResult.fromClauseCount == 1)
		{
			strncpy(tResult.tableName, tResult.fromClause[0].TableName, ZMDB_MAX_NAME_LEN);
		}
		return 0;
	}

    int SqlMap(const TSyntaxTree& oTree, ZmdbParseSqlStruct &tResult, TZmdbOrderBy* pOrderBy, int& OrderByCounts, TZmdbLimit* pLimit, TZmdbGroupBy* pGroupBy, string &sErrMsg)
    {
        int nRet = 0; 
        OrderByCounts = 0; 
        tResult.Clear();
        //memset(&tResult, 0, sizeof(tResult));

        switch(oTree.nOpType)
        {
            case OP_SELECT:
            	if (pOrderBy != NULL)
            	{
					for (int i = 0; i < ZMDB_MAX_SORT_COL_COUNTS; i++)
					{
						(pOrderBy + i)->Clear();
					}
            	}
            	if (pGroupBy != NULL)
            	{
					for (int i = 0; i < ZMDB_MAX_SORT_COL_COUNTS; i++)
					{
						(pGroupBy + i)->Clear();
					}
            	}
				if(pLimit != NULL)
				{
					pLimit->Clear();
				}
				else
				{
					return -1;
				}
				
                tResult.sqltype = OP_Query; 
                
				// add by jyhk at 2024-01-19, reason:避免SQL不带FROM字段不带表名
				if (oTree.u.pSelect->pTab == NULL)
				{
					sErrMsg = "查询语句必须使用'FROM'携带表名"; 
					MDB_CALL_ERR("Can't find table info on 'from' clause."); 
					return -1; 
				}

				//add by jyhk at 2022-12-19, reason:关联查询多表改造
				nRet = SelectFromMap(oTree.u.pSelect->pTab ,tResult, sErrMsg);
				if(0 != nRet)
				{
					MDB_CALL_ERR("SelectFromMap()"); 
					return nRet; 
				}

				nRet = OnMap(oTree.u.pSelect->pTab ,tResult, sErrMsg);
				if(0 != nRet)
				{
					MDB_CALL_ERR("OnMap()"); 
					return nRet; 
				}


				//add by jyhk at 2021-01-27, reason:for update nowait
				nRet = SelectOptionMap(oTree.u.pSelect->pOption ,tResult);
				if(0 != nRet)
				{
					MDB_CALL_ERR("SelectOptionMap()"); 
					return nRet; 
				}

				//匹配查询字段		
                nRet = SelectColumnMap(oTree.u.pSelect->pCol, oTree.u.pSelect->pGroupBy, tResult, sErrMsg, pOrderBy, OrderByCounts, pLimit); 
                if(0 != nRet)
                {
                    MDB_CALL_ERR("SelectColumnMap()"); 
                    return nRet; 
                }
				
				//匹配Where  语句
                nRet = WhereMap(oTree.u.pSelect->pWhere, tResult, sErrMsg); 
                if(0 != nRet)
                {
                    MDB_CALL_ERR("WhereMap()"); 
                    return nRet; 
                }
                
				//匹配GroupBy  语句--distinct + group by distinct就不用关心了
                nRet = GroupbyMap(oTree.u.pSelect->pGroupBy, pGroupBy, pOrderBy, OrderByCounts, tResult, sErrMsg); 
                if(0 != nRet)
                {
                    MDB_CALL_ERR("GroupbyMap()"); 
                    return nRet; 
                }
				if (pGroupBy[0].iCounts != 0)
				{
					tResult.tSelectType.bDistinct = false;
					tResult.tSelectType.bGroupby = true;
				}
				
				//匹配Order by  语句
				if (tResult.tSelectType.bDistinct)
				{//支持select distinct user_id  from userbill where user_id <= 100 order by user_id;的用法
					//排序字段必须保证在distinct字段之中存在
					nRet = DistinctOrderbyMap(oTree.u.pSelect->pOrderBy, pOrderBy, OrderByCounts, sErrMsg); 
					if(0 != nRet)
					{
						MDB_CALL_ERR("DistinctOrderbyMap()"); 
						return nRet; 
					}
					
				}
				else if (tResult.tSelectType.bGroupby)
				{//支持select user_id from userbill where user_id <= 100 group by user_id order by user_id;的用法
					//排序字段必须保证在group字段之中存在
					nRet = GroupOrderbyMap(oTree.u.pSelect->pOrderBy, pOrderBy, OrderByCounts, sErrMsg); 
					if(0 != nRet)
					{
						MDB_CALL_ERR("GroupOrderbyMap()"); 
						return nRet; 
					}
					
				}
				else
				{
					nRet = OrderbyMap(oTree.u.pSelect->pOrderBy, pOrderBy, OrderByCounts, sErrMsg); 
					if(0 != nRet)
					{
						MDB_CALL_ERR("OrderbyMap()"); 
						return nRet; 
					}
				}
				if (OrderByCounts != 0)
				{
					tResult.tSelectType.bSort = true;
				}
				
				//匹配limit  语句
				nRet = LimitMap(oTree.u.pSelect->pLimit, pLimit, tResult.tSelectType.bLimit, sErrMsg); 
				if(0 != nRet)
				{
					MDB_CALL_ERR("LimitMap()"); 
					return nRet; 
				}

				//检查查询语句正确性
				nRet = SelectColumnCheck(tResult);
				if(0 != nRet)
				{
					MDB_CALL_ERR("SelectColumnCheck()"); 
					return nRet; 
				}
                break; 
            case OP_RENAME:
                tResult.sqltype = OP_Rename; 
                strncpy(tResult.tableName, oTree.u.pRename->pTabSrc->szTabName,  sizeof(tResult.tableName));
                tResult.tableName[sizeof(tResult.tableName) -1] = '\0'; 
                strncpy(tResult.tableRename, oTree.u.pRename->pTabDst,  sizeof(tResult.tableRename));
                tResult.tableRename[sizeof(tResult.tableRename) -1] = '\0'; 
                break; 
            case OP_INSERT:
                tResult.sqltype = OP_Insert; 
                strncpy(tResult.tableName, oTree.u.pInsert->pTab->szTabName,  sizeof(tResult.tableName));
                tResult.tableName[sizeof(tResult.tableName) -1] = '\0'; 
                nRet = InsertColumnMap(oTree.u.pInsert, tResult, sErrMsg); 
                break; 
            case OP_UPDATE:
                tResult.sqltype = OP_Update; 
                strncpy(tResult.tableName, oTree.u.pUpdate->pTab->szTabName,  sizeof(tResult.tableName));
                tResult.tableName[sizeof(tResult.tableName) -1] = '\0'; 
                if (oTree.u.pUpdate->pTab->szTabAlias != NULL)
                {
					strncpy(tResult.tableAliasName, oTree.u.pUpdate->pTab->szTabAlias,	sizeof(tResult.tableAliasName));
					tResult.tableAliasName[sizeof(tResult.tableAliasName) -1] = '\0'; 
                }
                
                nRet = UpdateColumnMap(oTree.u.pUpdate->pCol, tResult, sErrMsg); 
                if(0 != nRet)
                {
                    MDB_CALL_ERR("UpdateColumnMap()"); 
                    return nRet; 
                }
                nRet = WhereMap(oTree.u.pUpdate->pWhere, tResult, sErrMsg); 
                if(0 != nRet)
                {
                    MDB_CALL_ERR("WhereMap()"); 
                    return nRet; 
                }

				// 稽核条件是否带别名
				for (int i = 0; i < tResult.whereClauseCount; i++)
				{
					if (tResult.whereClause[i].TableAliasName[0] != '\0')
					{
						if (TZmdbStrFunc::StrCmpNoCase(tResult.tableAliasName, tResult.whereClause[i].TableAliasName) != 0)
						{
							sErrMsg = "更新字段中表别名与操作表不匹配"; 
							MDB_ERR("%s", sErrMsg.c_str());
							return -1;
						}
					}
				}
                
                break; 
            case OP_DELETE:
                tResult.sqltype = OP_Delete; 
                strncpy(tResult.tableName, oTree.u.pDelete->pTab->szTabName,  sizeof(tResult.tableName));
                tResult.tableName[sizeof(tResult.tableName) -1] = '\0'; 
                if (oTree.u.pDelete->pTab->szTabAlias != NULL)
                {
					strncpy(tResult.tableAliasName, oTree.u.pDelete->pTab->szTabAlias,	sizeof(tResult.tableAliasName));
					tResult.tableAliasName[sizeof(tResult.tableAliasName) -1] = '\0';

                nRet = WhereMap(oTree.u.pDelete->pWhere,  tResult, sErrMsg); 
                if(0 != nRet)
                {
                    MDB_CALL_ERR("WhereMap()");

                }

				// 稽核条件是否带别名
				for (int i = 0; i < tResult.whereClauseCount; i++)
				{
					if (tResult.whereClause[i].TableAliasName[0] != '\0')
					{
						if (TZmdbStrFunc::StrCmpNoCase(tResult.tableAliasName, tResult.whereClause[i].TableAliasName) != 0)
						{
							sErrMsg = "更新字段中表别名与操作表不匹配"; 
							MDB_ERR("%s", sErrMsg.c_str());
							return -1;
						}
					}
				}
                break; 
            case OP_TRUNCATE:
                tResult.sqltype = OP_Truncate; 
                strncpy(tResult.tableName, oTree.u.pTrunc->pTab->szTabName,  sizeof(tResult.tableName));
                tResult.tableName[sizeof(tResult.tableName) -1] = '\0';

			case OP_DROPTABLE:
                tResult.sqltype = OP_DropTable; 
                strncpy(tResult.tableName, oTree.u.pDropTable->pTab->szTabName,  sizeof(tResult.tableName));
                tResult.tableName[sizeof(tResult.tableName) -1] = '\0';

					if ((int)strlen(oTree.u.pCreateTable->pTab->szTabName) >= ZMDB_MAX_NAME_LEN)
					{
					    sErrMsg = "create table error,table name over length, max length is 31!";
						return -1;
					}
                    strncpy(tResult.tableName, oTree.u.pCreateTable->pTab->szTabName,  sizeof(tResult.tableName));

    				tResult.columnCount = 0;
					tResult.extraColumnCount = 0;
    				TColumnList * pColumnList = oTree.u.pCreateTable->pCol;

						return -1;
					}
					bool bHasPrimaryKey = false;

    				{
    				    ColumnListItem & columnListItem = pColumnList->t[i];
    					tResult.columnOper[i].Clear();
                        if ((int)strlen(columnListItem.pName) >= ZMDB_MAX_NAME_LEN)
    					{
    					    sErrMsg = "create table error,column name over length, max length is 31!";
    						return -1;

    					strncpy(tResult.columnOper[i].ColumnName, columnListItem.pName, ZMDB_MAX_NAME_LEN);
    					tResult.columnOper[i].ColumnName[ZMDB_MAX_NAME_LEN - 1] = 0;
    					tResult.columnOper[i].valType = columnListItem.iItemType;
    					tResult.columnOper[i].valTypeLong = columnListItem.iItemLen;
						tResult.columnOper[i].bIsHighConflict = columnListItem.iIsHighConflict;
						if(columnListItem.pLoadColumnName != NULL)

						    if ((int)strlen(columnListItem.pLoadColumnName) >= ZMDB_MAX_CREATE_LOAD_COLUMN_LEN)

						if(tResult.columnOper[i].bIsPirmary == true)
						{

    				}
					if(bHasPrimaryKey == false)

					    sErrMsg = "Table must have primary key!";
						return -1;

					if(pCreateExtra != NULL)

						}

						TColumnList * pOrignColumnList = pAlterOption->pOrignColumnList;

						{
						    if(pOrignColumnList)
							{
							    char szErrMsg[128] = {0};
							    snprintf(szErrMsg, sizeof(szErrMsg), "Rename Column's count %d is wrong!", pOrignColumnList->nId);
							    sErrMsg = szErrMsg;
							}
							return -1;
						}
						ColumnListItem & columnListItem = pOrignColumnList->t[0];
    					tResult.columnOper[1].Clear();					
    					strncpy(tResult.columnOper[1].ColumnName, columnListItem.pName, ZMDB_MAX_NAME_LEN);
    					tResult.columnOper[1].ColumnName[ZMDB_MAX_NAME_LEN - 1] = 0;
    					tResult.columnOper[1].valType = columnListItem.iItemType;
    					tResult.columnOper[1].valTypeLong = columnListItem.iItemLen;
						if(columnListItem.pDefaultValue != NULL)
						{
							strncpy(tResult.columnOper[1].ColumnValue, columnListItem.pDefaultValue, MAX_COLUMNVALUE_SIZE);
							tResult.columnOper[1].ColumnValue[MAX_COLUMNVALUE_SIZE - 1] = 0;
						}
    					tResult.columnOper[1].bIsNullable = columnListItem.iIsNullable;
    					tResult.columnOper[1].bIsPirmary = columnListItem.iIsPrimaryKey;
						tResult.columnOper[1].alterType = pAlterOption->alterType;
    					tResult.columnCount++;
					}
                    break; 
				}
			case OP_DROPINDEX:
				{
					tResult.sqltype = OP_DropIndex; 
                    strncpy(tResult.tableName, oTree.u.pDropIndex->pTab->szTabName,  sizeof(tResult.tableName));
                    tResult.tableName[sizeof(tResult.tableName) -1] = '\0';
    				tResult.columnCount = 0;
					strncpy(tResult.tableRename, oTree.u.pDropIndex->pIndexName, ZMDB_MAX_NAME_LEN - 1);

					tResult.sqltype = OP_CreateIndex;

    				tResult.columnCount = 0;

					if ((int)strlen(oTree.u.pCreateIndex->pIndexName) >= ZMDB_MAX_NAME_LEN)

							}

    						return -1;
    					}

    					tResult.columnOper[i].valType = columnListItem.iItemType;
    					tResult.columnOper[i].valTypeLong = columnListItem.iItemLen;
						if(columnListItem.pDefaultValue != NULL)
						{
							strncpy(tResult.columnOper[i].ColumnValue, columnListItem.pDefaultValue, MAX_COLUMNVALUE_SIZE-1);
							tResult.columnOper[i].ColumnValue[MAX_COLUMNVALUE_SIZE - 1] = '\0';
						}
    					tResult.columnOper[i].bIsNullable = columnListItem.iIsNullable;
    					tResult.columnOper[i].bIsPirmary = columnListItem.iIsPrimaryKey;
    					tResult.columnCount++;
    				}
                    break; 
				}
			case OP_SETVAR:
    			{
				    tResult.sqltype = OP_SetVar; 
                    strncpy(tResult.tableName, oTree.u.pSetVar->pTab->szTabName,  sizeof(tResult.tableName));

    				tResult.columnCount = 0;
					TColumnList * pColumnList = oTree.u.pSetVar->pCol;
    				for(int i = 0; i < pColumnList->nId; ++i)
    				{
    				    ColumnListItem & columnListItem = pColumnList->t[i];

    					strncpy(tResult.columnOper[i].ColumnName, columnListItem.pName, ZMDB_MAX_NAME_LEN-1);
    					tResult.columnOper[i].ColumnName[ZMDB_MAX_NAME_LEN - 1] = '\0';
    					tResult.columnOper[i].valTypeLong = columnListItem.iItemLen;
						if(tResult.columnOper[i].valTypeLong <= 0 || tResult.columnOper[i].valTypeLong > 100)

						    sErrMsg = szErrMsg;
							return -1;
						}
    					tResult.columnCount++;
    				}
				    break;
    			}
            default:
                return -1; 
                break;
        }

        return nRet; 
    }

}//end of namespace


static bool IsFieldInList(const vector<TZmdbSlice> &vSliceFieldList, const char* pFieldName)
{
    for (unsigned int i = 0; i < vSliceFieldList.size(); ++i)
    {
        if (TZmdbStrFunc::StrCmpNoCase(vSliceFieldList[i].szSliceName, pFieldName) == 0)
        {
            return true;
        }
    }
    return false;
}

int CZmdbCommon::GetSqlType(const char*pSQL, int &nSqlType)
{
    int nRet = 0;
	CParser oParser; 

    //单纯的sql语法解析
    nRet = oParser.RunParser(pSQL);
    if(0 != nRet)
    {
        MDB_CALL_ERR("ZCParser::RunParser()");
        ZMDB_TADD_ERROR("syntax error! SQL[%s], errmsg[%s]", pSQL, oParser.GetLastErr());
        return SQLParseForSlice_Syntax_Error;
    }

    nSqlType = oParser.GetSyntax().nOpType;

    if (nSqlType == OP_NONE)
    {
        ZMDB_TADD_ERROR("syntax error! SQL[%s], errmsg[%s]", pSQL, "请检查SQL中是否包含不完整的引号");
        return SQLParseForSlice_Syntax_Error;
    }

    return 0;
}

int CZmdbCommon::AfterOn(std::map<std::string, TZmdbTableSliceInfo> &mapSliceFieldList, ZmdbParseSqlStruct& tTempStruct)
{
    set<std::string>setTables;
    for(int i=0; i<tTempStruct.onClauseCount; ++i)
    {
		//取左表检查是否是分片
        bool bLeftInList = IsFieldInList(mapSliceFieldList.find(tTempStruct.onClause[i].LeftTableAliasName)->second.vecSlice, tTempStruct.onClause[i].LeftColumnName);

		//取右表检查是否是分片
        bool bRightInList = IsFieldInList(mapSliceFieldList.find(tTempStruct.onClause[i].RightTableAliasName)->second.vecSlice, tTempStruct.onClause[i].RightColumnName);

		//下来了直接检查,只有左表和右表都使用分片才允许执行关联语句
		if (bLeftInList && bRightInList)
		{
			setTables.insert(tTempStruct.onClause[i].LeftTableAliasName);
			setTables.insert(tTempStruct.onClause[i].RightTableAliasName);
		}
    }   

	int iSliceTableCounts = 0;
    std::map<std::string, TZmdbTableSliceInfo>::iterator it = mapSliceFieldList.begin();
    for (; it != mapSliceFieldList.end(); it++)
    {
    	TZmdbTableSliceInfo &tZmdbTableSliceInfo = it->second;
		if (tZmdbTableSliceInfo.iMethodID != 0 && tZmdbTableSliceInfo.iMethodID != 5)
		{
			iSliceTableCounts++;
		}
    }

    //判断是否所有的分片字段都有,如果有分片字段没有引用到,则清空vector
	if ((int)setTables.size() != iSliceTableCounts)
	{
		//并非每个表都用到分片做关联关系,这个关联查询不能执行
		ZMDB_TADD_ERROR("Can't find all slice relation in join table!There has [%d] slice table ,but only has [%d] slice condition join on this table :", iSliceTableCounts, setTables.size());
		set<std::string>::iterator it = setTables.begin();
		while(it != setTables.end())
		{
			ZMDB_TADD_ERROR("Table[%s]", (*it).c_str());
			it++;
		}
		return -1;
	}

	return 0;
}

int CZmdbCommon::DealBeforeWhere(std::vector<TZmdbSlice> &vSliceFieldList, ZmdbParseSqlStruct& tTempStruct,
                                    vector<TZmdbSliceFieldInfo>& vFieldInfo)
{
    TZmdbSliceFieldInfo tSliceFieldInfo;
    
    for(int i=0; i<tTempStruct.columnCount; ++i)
    {
        bool bInList = IsFieldInList(vSliceFieldList, tTempStruct.columnOper[i].ColumnName);
        if (!bInList)
        {
            continue;
        }

        if (OP_Update == tTempStruct.sqltype)
        {
            ZMDB_TADD_ERROR("can't update slice field[%s]", tTempStruct.columnOper[i].ColumnName);
			//return -1;
        }
          
        //ZMDB_TADD_NORMAL("1--sParam=[%s]--->[%s]", pResult->columnOper[i].ColumnName, pResult->columnOper[i].ColumnValue);
        ZTMdbColumnOper tMdbColumnOper;
        tTempStruct.columnOper[i].To(tMdbColumnOper, true);

        memset(&tSliceFieldInfo, 0, sizeof(tSliceFieldInfo));
        strncpy(tSliceFieldInfo.ColumnName, tMdbColumnOper.sColumnName, ZMDB_MAX_NAME_LEN);
        strncpy(tSliceFieldInfo.ParamName, tMdbColumnOper.sParamName, ZMDB_MAX_NAME_LEN);
        if(tMdbColumnOper.sParamName[0] != 0)
        {
            tSliceFieldInfo.bBind = true;
        }
        else
        {
            tSliceFieldInfo.bBind = false;
			if (strlen(tMdbColumnOper.sColumnValue) > sizeof(tSliceFieldInfo.ColumnValue))
			{
				ZMDB_TADD_ERROR("Column lens[%d] too long.Column text = [%s].", strlen(tMdbColumnOper.sColumnValue), tMdbColumnOper.sColumnValue);
				return -1;
			}
            strncpy(tSliceFieldInfo.ColumnValue, tMdbColumnOper.sColumnValue, MAX_COLUMNVALUE_SIZE-1);
			tSliceFieldInfo.ColumnValue[MAX_COLUMNVALUE_SIZE-1] = '\0';
        }

        vFieldInfo.push_back(tSliceFieldInfo);
    }

    //给指定的分片字段改成绑定字段名
    for (unsigned int i = 0; i < vSliceFieldList.size(); ++i)
    {
        for (unsigned int j = 0; j < vFieldInfo.size(); ++j)
        {
            if (TZmdbStrFunc::StrCmpNoCase(vSliceFieldList[i].szSliceName, vFieldInfo[j].ColumnName) == 0)
            {
            	//如果插入字段使用绑定变量名，则赋值给分片字段；反之，直接把字段名赋值给分片字段
            	if (vFieldInfo[j].ParamName[0] != 0)
            	{
					strncpy(vSliceFieldList[i].szSliceAlias, vFieldInfo[j].ParamName, ZMDB_MAX_NAME_LEN);
            	}
            	else
            	{
					strncpy(vSliceFieldList[i].szSliceAlias, vFieldInfo[j].ColumnName, ZMDB_MAX_NAME_LEN);
            	}
                break;
            }
        }
    }    


    return 0;
}

//where条件中,需要有所有的分片字段,而且是and条件,才算引用到了分片索引
void CZmdbCommon::AfterWhere(std::vector<TZmdbSlice> &vSliceFieldList, ZmdbParseSqlStruct& tTempStruct,
                                    vector<TZmdbSliceFieldInfo>& vFieldInfo)
{
    bool bFound = false;
    TZmdbSliceFieldInfo tSliceFieldInfo;
    for(int i=0; i<tTempStruct.whereClauseCount; ++i)
    {
		// 转换条件对象
        ZTMdbWhereClause whereClause;
        tTempStruct.whereClause[i].To(whereClause);

		// 过滤二层条件, 二层条件不参与分片字段校验
		if (whereClause.iCurTag != 0)
		{
            continue;
		}

    	// OR 条件, 直接返回
        if (whereClause.iRelation == 2)   
        {
            vFieldInfo.clear();
            return;
        }

		// 非分片字段不校验
        bool bInList = IsFieldInList(vSliceFieldList, tTempStruct.whereClause[i].ColumnName);
        if (!bInList)
        {
            continue;
        }

        // 过滤非等式分片, 该逻辑不影响实际使用分片, 应该放入循环结束处的逻辑, 直接continue即可
        if (whereClause.cOperator != Cmp_equal)
        {
            continue;
        }

		// 过滤字段比较
		if (whereClause.bCompareWithColumn)
		{
			continue;
		}

		// 过滤取模
		if (whereClause.iUseMod == 1)
		{
			continue;
		}

		// add by jyhk at 2024-09-20, reason: 禁止重复字段进分片判断流程
		bool bFind = false;
		for (int j = 0; j < vFieldInfo.size(); j++)
		{
			if (TZmdbStrFunc::StrCmpNoCase(vFieldInfo[j].ColumnName, whereClause.sColumnName) == 0)
			{
				bFind = true;
				break;
			}
		}
		
		// mdy by jyhk at 2024-09-20, reason:过滤重复字段, 多个重复字段只看第一个
		if (bFind)
		{
			continue;
		}

		// 构造分片字段信息
		memset(&tSliceFieldInfo, 0, sizeof(tSliceFieldInfo));
		strncpy(tSliceFieldInfo.ColumnName, whereClause.sColumnName, ZMDB_MAX_NAME_LEN);
		strncpy(tSliceFieldInfo.ParamName, whereClause.sParamName, ZMDB_MAX_NAME_LEN);
		
		if(':' == tTempStruct.whereClause[i].CompareParam[0])
		{
			// 将绑定变量的分片字段放入数组		
			tSliceFieldInfo.bBind = true;
		}
		else
		{
			// 将非绑定变量的分片字段放入数组		
			tSliceFieldInfo.bBind = false;
			if (strlen(whereClause.sValue) > sizeof(tSliceFieldInfo.ColumnValue))
			{
				ZMDB_TADD_ERROR("Column lens[%d] too long.Column text = [%s].", strlen(whereClause.sValue), whereClause.sValue);
				return ;
			}
			strncpy(tSliceFieldInfo.ColumnValue, whereClause.sValue, MAX_COLUMNVALUE_SIZE-1);
			tSliceFieldInfo.ColumnValue[MAX_COLUMNVALUE_SIZE-1] = '\0';
		}

		vFieldInfo.push_back(tSliceFieldInfo);
    }   

    //判断是否所有的分片字段都有,如果有分片字段没有引用到,则清空vector
    for (unsigned int i = 0; i < vSliceFieldList.size(); ++i)
    {
        bFound = false;
        for (unsigned int j = 0; j < vFieldInfo.size(); ++j)
        {
            if (TZmdbStrFunc::StrCmpNoCase(vSliceFieldList[i].szSliceName, vFieldInfo[j].ColumnName) == 0)
            {
           		strncpy(vSliceFieldList[i].szSliceAlias, vFieldInfo[j].ParamName, ZMDB_MAX_NAME_LEN);
                bFound = true;
                break;
            }
        }

        if (!bFound)
        {
            vFieldInfo.clear();
            return;
        }
    }    
}

int CZmdbCommon::SQLParseTableName(const char*pSQL, TZMdbParseSliceField &tParseResult, CParser &oParser)
{
    int nRet = 0;

    //单纯的sql语法解析
    nRet = oParser.RunParser(pSQL);
    if(0 != nRet)
    {
        MDB_CALL_ERR("ZCParser::RunParser()");
        ZMDB_TADD_ERROR("syntax error! SQL[%s], errmsg[%s]", pSQL, oParser.GetLastErr());
        return SQLParseForSlice_Syntax_Error;
    }

    const TSyntaxTree &tSyntaxTree = oParser.GetSyntax();

    tParseResult.nSqlType = tSyntaxTree.nOpType;
    if (tParseResult.nSqlType == OP_RENAME)
    {
        //原表名
        strncpy(tParseResult.szTableName, tSyntaxTree.u.pRename->pTabSrc->szTabName,  ZMDB_MAX_NAME_LEN-1);
        tParseResult.szTableName[ZMDB_MAX_NAME_LEN-1] = '\0';
        //新表名
        strncpy(tParseResult.szTableNameDest, tSyntaxTree.u.pRename->pTabDst,  ZMDB_MAX_NAME_LEN-1);
        tParseResult.szTableName[ZMDB_MAX_NAME_LEN-1] = '\0';

        return SQLParseForSlice_OK;
    }
    
    if (tParseResult.nSqlType == OP_TRUNCATE)
    {
        strncpy(tParseResult.szTableName, tSyntaxTree.u.pTrunc->pTab->szTabName,  ZMDB_MAX_NAME_LEN-1);
        tParseResult.szTableName[ZMDB_MAX_NAME_LEN-1] = '\0';
        return SQLParseForSlice_OK;
    }

	if (tParseResult.nSqlType == OP_DROPTABLE)
    {
        strncpy(tParseResult.szTableName, tSyntaxTree.u.pDropTable->pTab->szTabName,  ZMDB_MAX_NAME_LEN-1);
        tParseResult.szTableName[ZMDB_MAX_NAME_LEN-1] = '\0';
        return SQLParseForSlice_OK;
    }
	if (tParseResult.nSqlType == OP_CREATETABLE)
    {
        strncpy(tParseResult.szTableName, tSyntaxTree.u.pCreateTable->pTab->szTabName,  ZMDB_MAX_NAME_LEN-1);
        tParseResult.szTableName[ZMDB_MAX_NAME_LEN-1] = '\0';
        return SQLParseForSlice_OK;
    }
	if(tParseResult.nSqlType == OP_ALTERTABLE)
	{
	    strncpy(tParseResult.szTableName, tSyntaxTree.u.pAlterTable->pTab->szTabName,  ZMDB_MAX_NAME_LEN-1);
        tParseResult.szTableName[ZMDB_MAX_NAME_LEN-1] = '\0';
        return SQLParseForSlice_OK;
	}

	if(tParseResult.nSqlType == OP_CREATEINDEX)
	{
	    strncpy(tParseResult.szTableName, tSyntaxTree.u.pCreateIndex->pTab->szTabName,  ZMDB_MAX_NAME_LEN-1);
        tParseResult.szTableName[ZMDB_MAX_NAME_LEN-1] = '\0';
        return SQLParseForSlice_OK;
	}

	if(tParseResult.nSqlType == OP_DROPINDEX)
	{
	    strncpy(tParseResult.szTableName, tSyntaxTree.u.pDropIndex->pTab->szTabName,  ZMDB_MAX_NAME_LEN-1);
        tParseResult.szTableName[ZMDB_MAX_NAME_LEN-1] = '\0';
        return SQLParseForSlice_OK;
	}

	if(tParseResult.nSqlType == OP_SETVAR)
	{
	    strncpy(tParseResult.szTableName, tSyntaxTree.u.pSetVar->pTab->szTabName,  ZMDB_MAX_NAME_LEN-1);
        tParseResult.szTableName[ZMDB_MAX_NAME_LEN-1] = '\0';
        return SQLParseForSlice_OK;
	}

    switch(tParseResult.nSqlType)
    {
    case OP_UPDATE:
        //取表名
        strncpy(tParseResult.szTableName, tSyntaxTree.u.pUpdate->pTab->szTabName,  ZMDB_MAX_NAME_LEN-1);
        tParseResult.szTableName[ZMDB_MAX_NAME_LEN -1] = '\0';
        break;
        
    case OP_SELECT:
		if (tSyntaxTree.u.pSelect->pTab == NULL)
		{
			ZMDB_TADD_ERROR("ZMDB SELECT must hava 'from' clause!"); 
			return -1;
		}
        
		/*mdy by jyhk at 2022-12-19, reason:关联查询多表改造*/
		if (tSyntaxTree.u.pSelect->pTab->nId > 1)
		{
			for (int i = 0; i < tSyntaxTree.u.pSelect->pTab->nId; i++)
			{
				TZmdbFromFieldInfo tFromFieldInfo;
				if (tSyntaxTree.u.pSelect->pTab->t[i].pTable->szTabAlias == NULL)
				{
					ZMDB_TADD_ERROR("Join query must use alias table name!"); 
					return -1;
				}
				
				strncpy(tFromFieldInfo.TableName, 
					tSyntaxTree.u.pSelect->pTab->t[i].pTable->szTabName, ZMDB_MAX_NAME_LEN);
				strncpy(tFromFieldInfo.AliasName, 
					tSyntaxTree.u.pSelect->pTab->t[i].pTable->szTabAlias, ZMDB_MAX_NAME_LEN);
				tParseResult.vTableInfo.push_back(tFromFieldInfo);
			}
		}
		else
		{
			//非关联查询暂时不动,DBC接口处新增关联查询标签,用于逻辑区分
			strncpy(tParseResult.szTableName, tSyntaxTree.u.pSelect->pTab->t[0].pTable->szTabName,  ZMDB_MAX_NAME_LEN-1);
			tParseResult.szTableName[ZMDB_MAX_NAME_LEN -1] = '\0';
		}
		
        break;
        
    case OP_DELETE:
        //取表名
        strncpy(tParseResult.szTableName, tSyntaxTree.u.pDelete->pTab->szTabName,  ZMDB_MAX_NAME_LEN-1);
        tParseResult.szTableName[ZMDB_MAX_NAME_LEN -1] = '\0';
        break; 
       
    case OP_INSERT:
        //取表名
        strncpy(tParseResult.szTableName, tSyntaxTree.u.pInsert->pTab->szTabName,  ZMDB_MAX_NAME_LEN-1);
        tParseResult.szTableName[ZMDB_MAX_NAME_LEN -1] = '\0';
        break;
        
    default:
        MDB_CALL_ERR("SQLParseTableName()");
        ZMDB_TADD_ERROR("unkown sqlType[%d],请检查SQL中是否包含不完整的引号", tParseResult.nSqlType);
        return SQLParseForSlice_Syntax_Error;
    }
    	
    return 0;
}

int CZmdbCommon::SQLParseSliceInfo(CParser &oParser, std::vector<TZmdbSlice> &vSliceFieldList, TZMdbParseSliceField &tParseResult, std::vector<std::string>& vecParamName, std::vector<std::string>& vecParamValue, 
								std::vector<int>& vecParamOper, std::vector<int>& vecParamRelation, std::vector<std::string>& vecPlaceHolder, string &sErrMsg)
{
    int nRet = 0;
    ZmdbParseSqlStruct *pTempStruct = NULL;

    const TSyntaxTree &tSyntaxTree = oParser.GetSyntax();

    tParseResult.nSqlType = tSyntaxTree.nOpType;
    if (tParseResult.nSqlType == OP_RENAME)
    {
        tParseResult.vColumnInfo.clear();
        tParseResult.vWhereInfo.clear();
        return SQLParseForSlice_OK;
    }
    
    if (tParseResult.nSqlType == OP_TRUNCATE)
    {
        tParseResult.vColumnInfo.clear();
        tParseResult.vWhereInfo.clear();
        return SQLParseForSlice_OK;
    }

	if (tParseResult.nSqlType == OP_DROPTABLE)
    {
        tParseResult.vColumnInfo.clear();
        tParseResult.vWhereInfo.clear();
        return SQLParseForSlice_OK;
    }

	if (tParseResult.nSqlType == OP_CREATETABLE)
    {
        tParseResult.vColumnInfo.clear();
        tParseResult.vWhereInfo.clear();
        return SQLParseForSlice_OK;
    }

	if (tParseResult.nSqlType == OP_ALTERTABLE)
    {
        tParseResult.vColumnInfo.clear();
        tParseResult.vWhereInfo.clear();
        return SQLParseForSlice_OK;
    }
    pTempStruct = new ZmdbParseSqlStruct();
    pTempStruct->Clear();
    //memset(pTempStruct, 0, sizeof(*pTempStruct));
    switch(tParseResult.nSqlType)
    {
    case OP_UPDATE:
        //取表名
        pTempStruct->sqltype = OP_Update; 
        strncpy(pTempStruct->tableName, tSyntaxTree.u.pUpdate->pTab->szTabName,  ZMDB_MAX_NAME_LEN-1);
        pTempStruct->tableName[ZMDB_MAX_NAME_LEN -1] = '\0';
        
		if (tSyntaxTree.u.pUpdate->pTab->szTabAlias != NULL)
		{
			strncpy(pTempStruct->tableAliasName, tSyntaxTree.u.pUpdate->pTab->szTabAlias,	ZMDB_MAX_NAME_LEN-1);
			pTempStruct->tableAliasName[ZMDB_MAX_NAME_LEN-1] = '\0'; 
		}

        //先处理set列表
        nRet = ZmdbSqlMap::UpdateColumnMap(tSyntaxTree.u.pUpdate->pCol, *pTempStruct, sErrMsg); 
        if(0 != nRet)
        {
            MDB_CALL_ERR("UpdateColumnMap()"); 
            return SQLParseForSlice_Update_Error; 
        }

        tParseResult.vColumnInfo.clear();
        nRet = CZmdbCommon::DealBeforeWhere(vSliceFieldList, *pTempStruct, tParseResult.vColumnInfo);
        if(0 != nRet)
        {
            MDB_CALL_ERR("DealBeforeWhere()"); 
            return SQLParseForSlice_Update_Error; 
        }

        //处理where条件
        nRet = ZmdbSqlMap::WhereMap(tSyntaxTree.u.pUpdate->pWhere, *pTempStruct, sErrMsg); 
        if(0 != nRet)
        {
            MDB_CALL_ERR("WhereMap()"); 
            return SQLParseForSlice_Where_Error; 
        }

		// 稽核条件是否带别名
		for (int i = 0; i < pTempStruct->whereClauseCount; i++)
		{
			if (pTempStruct->whereClause[i].TableAliasName[0] != '\0')
			{
				if (TZmdbStrFunc::StrCmpNoCase(pTempStruct->tableAliasName, pTempStruct->whereClause[i].TableAliasName) != 0)
				{
					MDB_CALL_ERR("WhereTableAlias()"); 
					return SQLParseForSlice_Where_Error; 
				}
			}
		}
        
        tParseResult.vWhereInfo.clear();
        CZmdbCommon::AfterWhere(vSliceFieldList, *pTempStruct, tParseResult.vWhereInfo);

		//add by jyhk at 2024-09-12, reason:获取所有IN占位符字段
		for (int i = 0; i < pTempStruct->whereClauseCount; ++i)
		{
			if (pTempStruct->whereClause[i].bPlaceHolder)
			{
				char szVarName[ZMDB_MAX_NAME_LEN] = {0};
				strncpy(szVarName, &pTempStruct->whereClause[i].CompareParam[1], ZMDB_MAX_NAME_LEN - 1);
				szVarName[ZMDB_MAX_NAME_LEN - 1] = '\0';
				vecPlaceHolder.push_back(szVarName);
			}
		}
        
        break;
        
    case OP_SELECT:
        //取表名
        pTempStruct->sqltype = OP_Query; 
        strncpy(pTempStruct->tableName, tSyntaxTree.u.pSelect->pTab->t[0].pTable->szTabName,  ZMDB_MAX_NAME_LEN-1);
        pTempStruct->tableName[ZMDB_MAX_NAME_LEN -1] = '\0';

        //处理where条件
        nRet = ZmdbSqlMap::WhereMap(tSyntaxTree.u.pSelect->pWhere, *pTempStruct, sErrMsg); 
        if(0 != nRet)
        {
            MDB_CALL_ERR("WhereMap()"); 
            return SQLParseForSlice_Where_Error; 
        }
        tParseResult.vWhereInfo.clear();
        CZmdbCommon::AfterWhere(vSliceFieldList, *pTempStruct, tParseResult.vWhereInfo);
		//拼出条件字段--供高速缓存使用
		for (int i = 0; i < pTempStruct->whereClauseCount; ++i)
		{
			vecParamName.push_back(pTempStruct->whereClause[i].ColumnName);
			vecParamValue.push_back(pTempStruct->whereClause[i].CompareParam);
			vecParamOper.push_back(pTempStruct->whereClause[i].Operator);
			vecParamRelation.push_back(pTempStruct->whereClause[i].relation);
		}

		//add by jyhk at 2024-09-12, reason:获取所有IN占位符字段
		for (int i = 0; i < pTempStruct->whereClauseCount; ++i)
		{
			if (pTempStruct->whereClause[i].bPlaceHolder)
			{
				char szVarName[ZMDB_MAX_NAME_LEN] = {0};
				strncpy(szVarName, &pTempStruct->whereClause[i].CompareParam[1], ZMDB_MAX_NAME_LEN - 1);
				szVarName[ZMDB_MAX_NAME_LEN - 1] = '\0';
				vecPlaceHolder.push_back(szVarName);
			}
		}


        break;
        
    case OP_DELETE:
        //取表名
        pTempStruct->sqltype = OP_Delete; 
        strncpy(pTempStruct->tableName, tSyntaxTree.u.pDelete->pTab->szTabName,  ZMDB_MAX_NAME_LEN-1);
        pTempStruct->tableName[ZMDB_MAX_NAME_LEN -1] = '\0';

		if (tSyntaxTree.u.pDelete->pTab->szTabAlias != NULL)
		{
			strncpy(pTempStruct->tableAliasName, tSyntaxTree.u.pDelete->pTab->szTabAlias,	ZMDB_MAX_NAME_LEN-1);
			pTempStruct->tableAliasName[ZMDB_MAX_NAME_LEN-1] = '\0'; 
		}        

        //处理where条件
        nRet = ZmdbSqlMap::WhereMap(tSyntaxTree.u.pDelete->pWhere, *pTempStruct, sErrMsg); 
        if(0 != nRet)
        {
            MDB_CALL_ERR("WhereMap()"); 
            return SQLParseForSlice_Where_Error; 
        }
        
		// 稽核条件是否带别名
		for (int i = 0; i < pTempStruct->whereClauseCount; i++)
		{
			if (pTempStruct->whereClause[i].TableAliasName[0] != '\0')
			{
				if (TZmdbStrFunc::StrCmpNoCase(pTempStruct->tableAliasName, pTempStruct->whereClause[i].TableAliasName) != 0)
				{
					MDB_CALL_ERR("WhereTableAlias()"); 
					return SQLParseForSlice_Where_Error; 
				}
			}
		}
        
        tParseResult.vWhereInfo.clear();
        CZmdbCommon::AfterWhere(vSliceFieldList, *pTempStruct, tParseResult.vWhereInfo);

        
		//add by jyhk at 2024-09-12, reason:获取所有IN占位符字段
		for (int i = 0; i < pTempStruct->whereClauseCount; ++i)
		{
			if (pTempStruct->whereClause[i].bPlaceHolder)
			{
				char szVarName[ZMDB_MAX_NAME_LEN] = {0};
				strncpy(szVarName, &pTempStruct->whereClause[i].CompareParam[1], ZMDB_MAX_NAME_LEN - 1);
				szVarName[ZMDB_MAX_NAME_LEN - 1] = '\0';
				vecPlaceHolder.push_back(szVarName);
			}
		}

        break; 
       
    case OP_INSERT:
        //取表名
        pTempStruct->sqltype = OP_Insert; 
        strncpy(pTempStruct->tableName, tSyntaxTree.u.pInsert->pTab->szTabName,  ZMDB_MAX_NAME_LEN-1);
        pTempStruct->tableName[ZMDB_MAX_NAME_LEN -1] = '\0';

        //处理insert列
        nRet = ZmdbSqlMap::InsertColumnMap(tSyntaxTree.u.pInsert, *pTempStruct, sErrMsg); 
        if(0 != nRet)
        {
            MDB_CALL_ERR("InsertColumnMap()"); 
            return SQLParseForSlice_Insert_Error; 
        }

        tParseResult.vColumnInfo.clear();
        nRet = CZmdbCommon::DealBeforeWhere(vSliceFieldList, *pTempStruct, tParseResult.vColumnInfo);
        if(0 != nRet)
        {
            MDB_CALL_ERR("DealBeforeWhere()"); 
            return SQLParseForSlice_Insert_Error; 
        }

        break;
        
    default:
        MDB_CALL_ERR("SQLParseForSlice()");
        ZMDB_TADD_ERROR("unkown sqlType[%d],请检查SQL中是否包含不完整的引号", tParseResult.nSqlType);
        return SQLParseForSlice_Syntax_Error;
    }
    if(pTempStruct != NULL)
    {
		delete pTempStruct;
		pTempStruct = NULL;
    }
    return 0;
}

int CZmdbCommon::SQLParseSpecialSelect(CParser &oParser, ZmdbParseSqlStruct &tTempStruct,ZMdbSelectType &tSelectType, TZmdbOrderBy* pOrderBy, int& iOrderByCounts,
												int &iLimitLeft, int &iLimitRight, TZmdbGroupBy* pGroupBy, int& iSpecialSelectColPos,string &sErrMsg)
{
    int nRet = 0;
    TZmdbLimit tLimit;
    //ZmdbParseSqlStruct tTempStruct;

    const TSyntaxTree &tSyntaxTree = oParser.GetSyntax();
    
	for (int i = 0; i < ZMDB_MAX_SORT_COL_COUNTS; i++)
	{
		if(pOrderBy != NULL)
		{
			(pOrderBy + i)->Clear();
		}
		else
		{
			return -1;
		}
	
		if(pGroupBy != NULL)
		{
			(pGroupBy + i)->Clear();
		}
		else
		{
			return -1;
		}
	}

    tSelectType.Clear();
	iOrderByCounts = 0;
    //memset(&tTempStruct, 0, sizeof(tTempStruct));
    tTempStruct.Clear();
    if (tSyntaxTree.nOpType == OP_SELECT)
    {
		//add by jyhk at 2021-01-27, reason:for update
		nRet = ZmdbSqlMap::SelectOptionMap(tSyntaxTree.u.pSelect->pOption ,tTempStruct);
		if(0 != nRet)
		{
			MDB_CALL_ERR("SelectOptionMap()"); 
			return nRet; 
		}
        //解析select 字段，主要是COUNT(*) 和SUM()
        nRet = ZmdbSqlMap::SelectColumnMap(tSyntaxTree.u.pSelect->pCol, tSyntaxTree.u.pSelect->pGroupBy, tTempStruct, sErrMsg, pOrderBy, iOrderByCounts, &tLimit); 
        if(0 != nRet)
        {
            MDB_CALL_ERR("SelectColumnMap()"); 
            return nRet; 
        }
		tSelectType = tTempStruct.tSelectType;

		/*add by jyhk at 2017-03-01, reason:新增给DBC接口用于group by*/
 		nRet = ZmdbSqlMap::GroupbyMap(tSyntaxTree.u.pSelect->pGroupBy, pGroupBy, pOrderBy, iOrderByCounts, tTempStruct, sErrMsg); 
		if(0 != nRet)
		{
			MDB_CALL_ERR("GroupbyMap()"); 
			return nRet; 
		}
		if (pGroupBy[0].iCounts != 0)
		{
			tSelectType.bDistinct = false;
			tSelectType.bGroupby = true;
		}

		/*add by jyhk at 2016-11-16, reason:新增给DBC接口用于排序*/
		
		//匹配Order by	语句
		if (tSelectType.bDistinct)
		{//支持select distinct user_id	from userbill where user_id <= 100 order by user_id;的用法
			//排序字段必须保证在distinct字段之中存在
			nRet = ZmdbSqlMap::DistinctOrderbyMap(tSyntaxTree.u.pSelect->pOrderBy, pOrderBy, iOrderByCounts, sErrMsg); 
			if(0 != nRet)
			{
				MDB_CALL_ERR("DistinctOrderbyMap()"); 
				return nRet; 
			}
		}
 		else if (tSelectType.bGroupby)
		{//支持select user_id from userbill where user_id <= 100 group by user_id order by user_id;的用法
			//排序字段必须保证在group字段之中存在
			nRet = ZmdbSqlMap::GroupOrderbyMap(tSyntaxTree.u.pSelect->pOrderBy, pOrderBy, iOrderByCounts, sErrMsg); 
			if(0 != nRet)
			{
				MDB_CALL_ERR("GroupOrderbyMap()"); 
				return nRet; 
			}
		}
		else
		{
			nRet = ZmdbSqlMap::OrderbyMap(tSyntaxTree.u.pSelect->pOrderBy, pOrderBy, iOrderByCounts, sErrMsg); 
			if(0 != nRet)
			{
				MDB_CALL_ERR("OrderbyMap()"); 
				return nRet; 
			}
		}
		if (iOrderByCounts != 0)
		{
			tSelectType.bSort = true;
		}

		//group + max\min + limit场景解析
		if (tSelectType.bGroupby)
		{
			tSelectType.bLimit = false;
			/*add by jyhk at 2016-11-16, reason:新增给DBC接口用于Limit*/
			nRet = ZmdbSqlMap::LimitMap(tSyntaxTree.u.pSelect->pLimit, &tLimit, tSelectType.bLimit, sErrMsg); 
			if(0 != nRet)
			{
				MDB_CALL_ERR("LimitMap()"); 
				return nRet; 
			}
		}
		else
		{
			if (tSelectType.bMax || tSelectType.bMin)
			{//max , min自带limit
				//do nothing
			}
			else
			{//只有limit的场景
				/*add by jyhk at 2016-11-16, reason:新增给DBC接口用于Limit*/
				nRet = ZmdbSqlMap::LimitMap(tSyntaxTree.u.pSelect->pLimit, &tLimit, tSelectType.bLimit, sErrMsg); 
				if(0 != nRet)
				{
					MDB_CALL_ERR("LimitMap()"); 
					return nRet; 
				}
			}
		}

		if (tSelectType.bLimit)
		{
			iLimitLeft = tLimit.iLeftLimitNum;
			iLimitRight = tLimit.iRightLimitNum;
		}
		else
		{
			iLimitLeft = 0;
			iLimitRight = 0;
		}

		iSpecialSelectColPos = -1;
		for (int i = 0; i < tTempStruct.columnCount; i++)
		{
			if (tTempStruct.columnOper[i].bCount || tTempStruct.columnOper[i].bSum 
					|| tTempStruct.columnOper[i].bMax|| tTempStruct.columnOper[i].bMin)
			{
				iSpecialSelectColPos = i;
			}
		}

		//add by jyhk at 2022-12-22, reason:关联查询支持
		if (tSyntaxTree.u.pSelect->pTab->nId > 1)
		{
			tSelectType.bJoin = true;
		}

		nRet = SelectColumnCheckInDBC(tSelectType, sErrMsg);
		if(0 != nRet)
		{
			MDB_CALL_ERR("SelectColumnCheckInDBC()"); 
			return nRet; 
		}
    }
    
    return 0;

}

long long CZmdbCommon::GetHashValue(char *pDataBuf, int iPos, int iDataType, int iDataLen)
{
    char *pData = NULL;
    //double dData = 0;
    long long llData = 0;
    time_t tVale = 0;
    char szData[MAX_BLOB_LEN] = {0};

    //pData = (char*)(m_tSetParamData.Data()) + iPos;

    pData = pDataBuf+iPos;
    
    switch (iDataType)
    {
        case DT_Int:
            switch(iDataLen)
            {
                case 1:
                    llData = *(char*)pData;
                    break;
                case 2:
                    llData = *(short*)pData;
                    break;
                case 4:
                    llData = *(int*)pData;
                    break;
                case 8:
                    llData = *(long long*)pData;
                    break;
            }
            break;
            
        case DT_VarChar:
        case DT_Char:
            strncpy(szData, pData, iDataLen);
            llData = TZmdbStrFunc::StrToHash(szData);
            break;

        case DT_DateStamp:
            if (iDataLen == sizeof(time_t))
            {
				//llData = atol(pData); mdy by jyhk , reason：目前不会有这样的数据进来,即以字符串的形式绑定time_t
                llData = *(time_t*)pData;
            }
            else
            {
                if (MAX_TIME_LEN != iDataLen-1)
                {
                    //ERROR
                }
                strncpy(szData, pData, iDataLen);
                mdbFormatStrToTime(szData, tVale);
                llData = tVale;
            }
            break;

        case DT_Double:
            /*
            dData = *(double*)pData;
            *(double *)&llData = dData;*/
            memcpy((void*)&llData, (void*)pData, sizeof(llData));
            break;

        default:
            break;
    }

    return llData;
}

int CZmdbCommon::CaculateHash(int nTimes ,unsigned long long& llValue1,unsigned long long llValue2)
{
    if (nTimes == 0)//第一个字段直接赋值
    {
        llValue1 = llValue2;
    }
    else
    {
        llValue1  = (llValue1 << ((long long)(floor(log(double(llValue2))/log(2.0))+1))) | llValue2;
    }
    return 0;
}

long long CZmdbCommon::GenTransIdByConnId(int iConnId)
{
    long long llTransID=0;
    char szCurrDate[MAX_TIME_LEN]={0};
    
    TZmdbDateTimeFunc::GetCurrentTimeStr(szCurrDate,MAX_TIME_LEN);
    
    szCurrDate[MAX_TIME_LEN-3] = '\0'; //去掉秒(SS)
    llTransID = atol(szCurrDate+4) * 100000000000  + iConnId; //去掉年(YYYY)

    ZMDB_TADD_DETAIL("llTransID=%lld", llTransID);

    return llTransID;
}

/*add by jyhk at 2017-03-30, reason:新增给高速缓存进行解析用*/
int CZmdbCommon::SQLParseForCache(const TSyntaxTree& oTree, ZmdbParseSqlStruct &tResult, string &sErrMsg)
{
	int nRet = 0; 
	//memset(&tResult, 0, sizeof(tResult));
	tResult.Clear();
	TZmdbOrderBy* pOrderBy = new TZmdbOrderBy[ZMDB_MAX_SORT_COL_COUNTS];
	TZmdbLimit tLimit;
	int OrderByCounts;

	//匹配操作-----  缓存目前只支持查询
	tResult.sqltype = OP_Query; 

	//匹配表名
	strncpy(tResult.tableName, oTree.u.pSelect->pTab->t[0].pTable->szTabName,  sizeof(tResult.tableName));
	tResult.tableName[sizeof(tResult.tableName) -1] = '\0'; 
	
	//匹配查询字段		
	nRet = ZmdbSqlMap::SelectColumnMap(oTree.u.pSelect->pCol, oTree.u.pSelect->pGroupBy, tResult, sErrMsg, pOrderBy, OrderByCounts, &tLimit); 
	if(0 != nRet)
	{
		MDB_CALL_ERR("SelectColumnMap()"); 
		return nRet; 
	}
	
	//匹配Where  语句
	nRet = ZmdbSqlMap::WhereMap(oTree.u.pSelect->pWhere, tResult, sErrMsg); 
	if(0 != nRet)
	{
		MDB_CALL_ERR("WhereMap()"); 
		return nRet; 
	}

	delete []pOrderBy;
	pOrderBy = NULL;
	return 0;
}

/*add by jyhk at 2017-02-17, reason:特殊查询校验*/
int CZmdbCommon::SelectColumnCheckInDBC(ZMdbSelectType &tSelectType, string &sErrMsg)
{

	if(true == tSelectType.bGroupby)
	{
		if (true == tSelectType.bAll)
		{
			sErrMsg =  "[GroupBy] and [*] can't be used simultaneously!"; 
			MDB_ERR("%s", sErrMsg.c_str()); 
			return -1; 
		}
	}
	else if (true == tSelectType.bSort && false == tSelectType.bMax && false == tSelectType.bMin)//order by and not group by
	{
		if(true == tSelectType.bCount && false == tSelectType.bDistinct)
		{
			sErrMsg =  "排序时不支持使用Count!没有使用意义,请直接使用Count即可!";
			MDB_ERR("%s", sErrMsg.c_str()); 
			return -1; 
		}

		if(true == tSelectType.bSum)
		{
			sErrMsg =  "排序时不支持使用Sum!没有使用意义,请直接使用Sum即可!";
			MDB_ERR("%s", sErrMsg.c_str()); 
			return -1; 
		}

		if(true == tSelectType.bMax)
		{
			sErrMsg =  "排序时不支持使用Max!没有使用意义,请直接使用Max即可!";
			MDB_ERR("%s", sErrMsg.c_str()); 
			return -1; 
		}

		if(true == tSelectType.bMin)
		{
			sErrMsg =  "排序时不支持使用Min!没有使用意义,请直接使用Min即可!";
			MDB_ERR("%s", sErrMsg.c_str()); 
			return -1; 
		}
	}

	//关联查询特殊查询组合限制
	if (true == tSelectType.bJoin)
	{
		if (true == tSelectType.bGroupby)
		{
			sErrMsg =  "关联查询暂时不支持分组!"; 
			MDB_ERR("%s", sErrMsg.c_str()); 
			return -1; 
		}

		if (true == tSelectType.bSort ||true == tSelectType.bDistinct|| true == tSelectType.bMax|| true == tSelectType.bMin)
		{
			sErrMsg =  "关联查询暂时不支持排序!"; 
			MDB_ERR("%s", sErrMsg.c_str()); 
			return -1; 
		}

		if (true == tSelectType.bSum)
		{
			sErrMsg =  "关联查询暂时不支持聚合函数!"; 
			MDB_ERR("%s", sErrMsg.c_str()); 
			return -1; 
		}

		if (true == tSelectType.bForupdate)
		{
			sErrMsg =  "关联查询暂时不支持加锁!"; 
			MDB_ERR("%s", sErrMsg.c_str()); 
			return -1; 
		}
	}

	return 0;
}

int CZmdbCommon::SQLParseJoinSliceInfo(CParser &oParser, std::map<std::string, TZmdbTableSliceInfo> &mapSliceFieldList, TZMdbParseSliceField &tParseResult, std::vector<std::string>& vecParamName, std::vector<std::string>& vecParamValue, std::vector<int>& vecParamOper, std::vector<int>& vecParamRelation, std::vector<std::string>& vecPlaceHolder, string &sErrMsg, bool bCheckJoinSlice)
{
    int nRet = 0;

	char szErrmsg[ZMDB_MAX_ERR_LEN] = {0};
    
    ZmdbParseSqlStruct *pTempStruct = NULL;

    const TSyntaxTree &tSyntaxTree = oParser.GetSyntax();

    tParseResult.nSqlType = tSyntaxTree.nOpType;

    pTempStruct = new ZmdbParseSqlStruct();
    //memset(pTempStruct, 0, sizeof(*pTempStruct));
    pTempStruct->Clear();

    //取表名
    pTempStruct->sqltype = OP_Query; 

    //取表数量
    pTempStruct->fromClauseCount = tSyntaxTree.u.pSelect->pTab->nId;

    //处理where条件
    nRet = ZmdbSqlMap::WhereMap(tSyntaxTree.u.pSelect->pWhere, *pTempStruct, sErrMsg); 
    if(0 != nRet)
    {
        MDB_CALL_ERR("WhereMap()"); 
        return SQLParseForSlice_Where_Error; 
    }
    
    tParseResult.vWhereInfo.clear();
    tParseResult.vOnInfo.clear();

    /*mdy by jyhk at 20222-12-22, reason:描述on关系中是否有用到分片*/
    nRet = ZmdbSqlMap::OnMap(tSyntaxTree.u.pSelect->pTab, *pTempStruct, sErrMsg); 
    if(0 != nRet)
    {
        MDB_CALL_ERR("OnMap()"); 
        return -1; 
    }

	//检查关联关系是否建立到表上
	for(int i=0; i<pTempStruct->onClauseCount; ++i)
	{
		if (mapSliceFieldList.find(pTempStruct->onClause[i].LeftTableAliasName) == mapSliceFieldList.end())
		{
			snprintf(szErrmsg, ZMDB_MAX_ERR_LEN, "Table alias[%s] not in from list.", pTempStruct->onClause[i].LeftTableAliasName);
			sErrMsg = szErrmsg;
			ZMDB_TADD_ERROR(szErrmsg); 
			return -1;
		}
		else if (mapSliceFieldList.find(pTempStruct->onClause[i].RightTableAliasName) == mapSliceFieldList.end())
		{
			snprintf(szErrmsg, ZMDB_MAX_ERR_LEN, "Table alias[%s] not in from list.", pTempStruct->onClause[i].RightTableAliasName);
			sErrMsg = szErrmsg;
			ZMDB_TADD_ERROR(szErrmsg); 
			return -1;
		}
	}

	int iSliceTableCounts = 0;
    std::map<std::string, TZmdbTableSliceInfo>::iterator it = mapSliceFieldList.begin();
    for (; it != mapSliceFieldList.end(); it++)
    {
    	TZmdbTableSliceInfo &tZmdbTableSliceInfo = it->second;
		if (tZmdbTableSliceInfo.iMethodID != 0 && tZmdbTableSliceInfo.iMethodID != 5)
		{
			iSliceTableCounts++;
		}
    }

	//mdy by lijing at 2024-04-17, reason:先判断是否需要校验分片
	if (bCheckJoinSlice)
	{
		//解析关联关系是否建立在分片字段上
		if (iSliceTableCounts > 1)//当且仅当使用分片的表大于1个时，才进行多表关联关系判断
		{
			if (mapSliceFieldList.begin()->second.vecSlice.size() != 0)
			{
				nRet = CZmdbCommon::AfterOn(mapSliceFieldList, *pTempStruct);
				if(0 != nRet)
				{
					sErrMsg = "Can't find all slice relation in join table!";
					MDB_CALL_ERR("AfterOn()"); 
					return -1; 
				}
			}
		}
	}

	//判断是否只有一个表带分片条件
	std::vector<TZmdbSliceFieldInfo> vWhereInfo;
    for (int i = 0; i < tSyntaxTree.u.pSelect->pTab->nId; i++)
    {
    	vWhereInfo.clear();
		CZmdbCommon::AfterJoinWhere(tSyntaxTree.u.pSelect->pTab->t[i].pTable->szTabAlias, mapSliceFieldList, *pTempStruct, vWhereInfo);
		if (vWhereInfo.size() != 0)
		{
			//解析分片条件
			if (tParseResult.vWhereInfo.size() == 0)
			{
				tParseResult.vWhereInfo.assign(vWhereInfo.begin(), vWhereInfo.end());
				strncpy(tParseResult.szTableName, tSyntaxTree.u.pSelect->pTab->t[i].pTable->szTabAlias, ZMDB_MAX_NAME_LEN - 1);
				tParseResult.szTableName[ZMDB_MAX_NAME_LEN - 1] = '\0';
			}
			else
			{
				//只允许一个表带分片
				sErrMsg = "Join query not support multi slice condition!";
				ZMDB_TADD_ERROR("Join query not support multi slice condition!"); 
				return -1;
			}
		}
    }

	//判断是否每个表都有关联关系
    for (int i = 0; i < tSyntaxTree.u.pSelect->pTab->nId; i++)
    {
    	int j=0;
		for(; j<pTempStruct->onClauseCount; ++j)
		{
			if (0 == TZmdbStrFunc::StrCmpNoCase(pTempStruct->onClause[j].LeftTableAliasName, tSyntaxTree.u.pSelect->pTab->t[i].pTable->szTabAlias) ||
				0 == TZmdbStrFunc::StrCmpNoCase(pTempStruct->onClause[j].RightTableAliasName, tSyntaxTree.u.pSelect->pTab->t[i].pTable->szTabAlias))
			{
				break;
			}
		}	
		if (j == pTempStruct->onClauseCount)
		{
			snprintf(szErrmsg, ZMDB_MAX_ERR_LEN, "Table[%s] can't find join relation!", tSyntaxTree.u.pSelect->pTab->t[i].pTable->szTabName);
			sErrMsg = szErrmsg;
			ZMDB_TADD_ERROR(szErrmsg); 
			return -1;
		}
    }
	
	//add by jyhk at 2024-09-12, reason:获取所有IN占位符字段
	for (int i = 0; i < pTempStruct->whereClauseCount; ++i)
	{
		if (pTempStruct->whereClause[i].bPlaceHolder)
		{
			char szVarName[ZMDB_MAX_NAME_LEN] = {0};
			strncpy(szVarName, &pTempStruct->whereClause[i].CompareParam[1], ZMDB_MAX_NAME_LEN - 1);
			szVarName[ZMDB_MAX_NAME_LEN - 1] = '\0';
			vecPlaceHolder.push_back(szVarName);
		}
	}
  
    if(pTempStruct != NULL)
    {
		delete pTempStruct;
		pTempStruct = NULL;
    }
    return 0;
}

//where条件中,需要有所有的分片字段,而且是and条件,才算引用到了分片索引
void CZmdbCommon::AfterJoinWhere(char* pszTableAlias, std::map<std::string, TZmdbTableSliceInfo> &mapSliceFieldList, ZmdbParseSqlStruct& tTempStruct,
                                    vector<TZmdbSliceFieldInfo>& vFieldInfo)
{
    bool bFound = false;

	std::vector<TZmdbSlice> &vSliceFieldList = mapSliceFieldList.find(pszTableAlias)->second.vecSlice;
    
    TZmdbSliceFieldInfo tJoinSliceFieldInfo;
    for(int i=0; i<tTempStruct.whereClauseCount; ++i)
    {
        if (tTempStruct.whereClause[i].relation == 2)   // OR 条件, 直接返回
        {
            vFieldInfo.clear();
            return;
        }

		if (0 != strcmp(pszTableAlias, tTempStruct.whereClause[i].TableAliasName))
		{
			continue;
		}

        bool bFieldInList = IsFieldInList(vSliceFieldList, tTempStruct.whereClause[i].ColumnName);
        if (!bFieldInList)
        {
            continue;
        }

		//add by jyhk at 2019-05-01,reason:当这个分片是在MOD的表达式中时,这次操作就认为不使用分片
        if (1 == tTempStruct.whereClause[i].iUseMod)
        {
            vFieldInfo.clear();
            return;
        }

        ZTMdbWhereClause whereClause;
        tTempStruct.whereClause[i].To(whereClause);
        if (whereClause.cOperator != Cmp_equal)     // 不是 = 关系, 直接返回
        {
            vFieldInfo.clear();
            return;
        }

        memset(&tJoinSliceFieldInfo, 0, sizeof(tJoinSliceFieldInfo));
        strncpy(tJoinSliceFieldInfo.ColumnName, whereClause.sColumnName, ZMDB_MAX_NAME_LEN);
        strncpy(tJoinSliceFieldInfo.ParamName, whereClause.sParamName, ZMDB_MAX_NAME_LEN);
        if(':' == tTempStruct.whereClause[i].CompareParam[0])
        {
            tJoinSliceFieldInfo.bBind = true;
        }
        else
        {
            tJoinSliceFieldInfo.bBind = false;
			if (strlen(whereClause.sValue) > sizeof(tJoinSliceFieldInfo.ColumnValue))
			{
				ZMDB_TADD_ERROR("Column lens[%d] too long.Column text = [%s].", strlen(whereClause.sValue), whereClause.sValue);
				return ;
			}
            strncpy(tJoinSliceFieldInfo.ColumnValue, whereClause.sValue, MAX_COLUMNVALUE_SIZE-1);
			tJoinSliceFieldInfo.ColumnValue[MAX_COLUMNVALUE_SIZE-1] = 0;
        }

        vFieldInfo.push_back(tJoinSliceFieldInfo);
    }   

    //判断是否所有的分片字段都有,如果有分片字段没有引用到,则清空vector
    for (unsigned int i = 0; i < vSliceFieldList.size(); ++i)
    {
        bFound = false;
        for (unsigned int j = 0; j < vFieldInfo.size(); ++j)
        {
            if (TZmdbStrFunc::StrCmpNoCase(vSliceFieldList[i].szSliceName, vFieldInfo[j].ColumnName) == 0)
            {
           		strncpy(vSliceFieldList[i].szSliceAlias, vFieldInfo[j].ParamName, ZMDB_MAX_NAME_LEN);
                bFound = true;
                break;
            }
        }

        if (!bFound)
        {
            vFieldInfo.clear();
            return;
        }
    }    
}
